A 题目名称：

A Bad Ugly Numbers

题目大意：

构造一个 n位十进制数，保证不被该数的任意一位整除。

题解：

由于只需要求一组即可，那么就猜测一下一个数字一定要么是奇数要么是偶数，若为奇数则个位不为偶数，若为偶数则个位为偶数。这么不妨用2来替换。假如剩余全为3，那么3*x+2一定无法被3整除。所以就全用2，3表示。偶数不能被2整除。那么开头数字为2，其余全为3就可以了。

代码：

#include<bits/stdc++.h>
using namespace std;
int T,n;
int main()
{
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d",&n);
        if(n==1)
        {
            printf("-1\n");
        }
        else 
        {
            printf("2");
            for(int i=1;i<=n-1;i++)
            printf("3");
            cout<<endl;
        }
    }
    return 0;
}

B 题目名称：

Maximums

题目大意：

对于一个序列 {an}，定义序列 {cn} ：ci=max{0,a1,⋯,ai−1} 。同时定义序列 {bn} :bi=ai−ci 。

给定 {bn}，求 {an} 。

题解：

由题意可知：ci=max{0,a1,…,ai-1},bi=ai-ci.所以ai=bi+ci;那么维护max{0,a1,…ai-1}即可。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=200000 +5;
long long n,b[maxn],a[maxn];
int main()
{
    scanf("%lld",&n);
    for(long long i=1;i<=n;i++)scanf("%lld",&b[i]);
    long long maxx=b[1];
    a[1]=b[1];
    for(long long i=2;i<=n;i++)
    {
        a[i]=maxx+b[i];
        maxx=max(maxx,a[i]);
    }
    for(long long i=1;i<=n;i++)
    printf("%lld ",a[i]);
    cout<<endl;
    return 0;
}

C 题目名称：

Permutation Partitions

题目大意：

给定一个长度为 n的排列{an}。要求将这个序列分成互不相交的 k段。记第 p段的左端点和右端点分别为 lp,rp 。要求最大化
输出最大化的值和可以最大化该值的方案数。方案数对 998,244,353取模。

题解：

将长度为n的排列分成互不相交的k段，并且取每段的最大值，问它们的和。这一问就直接找到前k大的数字，并求和就可以了。然后求有多少种方案数。这一问需要看一下规律，将前k大的数字进行标记，然后依次统计相邻标记的数字中间有几个未标记的位置，如果为0，这区域的方案数为，如果为1，这区域的方案数位2，如果为2，这区域的方案数为3，如果为k，这区域的方案数为k+1。最后全部一边成一边取模。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=200000 +5;
const int Mod=998244353;
int n,m,a[maxn],d[maxn],vis[maxn],tot,f[maxn];long long ans=1,sig=0;
int main()
{
    scanf("%d%d",&n,&m);
    f[0]=1;
    for(int i=1;i<=n;i++)
    {
        f[i]=i+1;
    }
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
        if(a[i]>=n-m+1&&a[i]<=n)
        {
            sig+=a[i];
            vis[i]=1;
        }
    }
    int maxx=0;
    for(int i=1;i<=n;i++)
    {
        if(maxx==0&&vis[i]==1)
        {
            maxx=i;
        }
        else if(vis[i]==1)
        {
            d[++tot]=i-maxx-1;
            maxx=i;
        }
    }
    ans=f[d[1]];
    for(int i=2;i<=tot;i++)
    {
        ans=(ans*f[d[i]])%Mod;
    }
    printf("%lld %lld\n",sig,ans);
    return 0;
}

D 题目名称：

题目大意：

给定一个字符串。要求选取他的一个前缀(可以为空)和与该前缀不相交的一个后缀(可以为空)拼接成回文串，且该回文串长度最大。求该最大长度。

题解：

Easy版本：

这个数据范围更小，T<=1000,n<=5000；可以暴力。首先从两边往中间收，一一匹配，直到左右不等，记录opt1，opt2。然后看以opt1为起点最长的回文串，然后以opt2为起点最长的回文串，这里可以用暴力办法，比如以opt1为起点时，枚举终点，从opt2开始，一旦中间不符合了，就终点位置往前靠一位；同理以opt2为起点时，枚举终点，从opt1开始，一旦中间不符合了，终点位置就往后靠一位。最后选择一个最长的与开始的合并输出。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=50000+5;
int n,l;char s[maxn],t[maxn];
int main()
{
    scanf("%d",&n);
    while(n--)
    {
        memset(t,0,sizeof(t));l=0;
        scanf("%s",s+1);
        int opt1=1,opt2=strlen(s+1);
        while(opt1<=opt2)
        {
            if(s[opt1]==s[opt2])
            {
                t[++l]=s[opt1];
                opt1++;
                opt2--;
            }
            else 
            break;
        }
        if(l==(strlen(s+1)+1)/2)
        {
            int len=strlen(s+1);
            for(int i=1;i<=strlen(s+1);i++)
            cout<<s[i];
            cout<<endl;
            continue;
        }
        int s1=opt1,s2=opt2,tot1=0,sign1=0,T1=0,c=1;
        while(s1<=s2)
        {
            if(s1==s2)sign1=1;
            if(s[s1]==s[s2])
            {
                s1++;s2--;
                tot1++;
                T1+=2;
            }
            else 
            {
                tot1=0;
                s1=opt1;s2=opt2-c;//WA
                c++;
                T1=0;
            }
        }
        if(sign1==1)T1--;
        s1=opt1;s2=opt2;int tot2=0,sign2=0,T2=0;c=1;
        while(s1<=s2)
        {
            if(s1==s2)sign2=1;
            if(s[s1]==s[s2])
            {
                s1++;s2--;
                tot2++;
                T2+=2;
            }
            else 
            {
                s1=opt1+c;s2=opt2;c++;
                tot2=0;
                T2=0;
            }
        }
        if(sign2==1)T2--;
        if(T1>T2)
        {
            for(int i=opt1;i<=opt1+tot1-1;i++)
            {
                t[++l]=s[i];
            }
            if(sign1==1)
            {
                int len=l;
                for(int i=len-1;i>=1;i--)t[++l]=t[i];
            }
            else 
            {
                int len=l;
                for(int i=len;i>=1;i--)t[++l]=t[i];
            }
        }
        else 
        {
            for(int i=opt2;i>=opt2-tot2+1;i--)
            t[++l]=s[i];
            if(sign2==1)
            {
                int len=l;
                for(int i=len-1;i>=1;i--)t[++l]=t[i];
            }
            else 
            {
                int len=l;
                for(int i=len;i>=1;i--)t[++l]=t[i];
            }
        }
        cout<<t+1<<endl;
    }
    return 0;
}
hard版本：

这里数据范围很大，T<=100000,n<=1000000.显然每次查询必须在O(n)内完成，所以必须用到Manacher的算法了。首先还是先从两端开始匹配出相同的总数，直到两端不等为止，记录opt1，opt2。然后以opt1为起点求最长的回文串，再以opt2为起点求最长的回文串。这里要注意Manacher求得的最长回文串可能不一定以opt1为起点。所以更新长度时候要注意让i-(p[i]-1)==1然后再更新长度。当以opt2为起点时候，将字符串s，从opt1到opt2之间翻转一下reverse(s+opt1,s+1+opt2)即可。然后选择最长的回文串与开始的合并输出。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=3000000+5;
int T,p[maxn],l;char s[maxn]; 
int Manacher(int S,int T)
{
    int len=1;
    string ss="$#";
    for(int i=S;i<=T;i++)
    {
        ss+=s[i];ss+='#';
        len+=2;
    }
    int mx=0,id=0,ans=0;p[0]=p[1]=0;
    for(int i=1;i<=len;i++)
    {
        if(i<mx)p[i]=min(p[id*2-i],mx-i);
        else p[i]=1;
        while(ss[i+p[i]]==ss[i-p[i]])p[i]++;
        if(i+p[i]>mx)
        {
            mx=i+p[i];
            id=i;
        }
        ans=max(ans,p[i]-1);
        if(i-p[i]+1==1)
        {
            if(l<p[i]-1)
            {
                l=p[i]-1;
            }
        }
    }
    return l;
}
int main()
{
    scanf("%d",&T);
    while(T--)
    {
        scanf("%s",s+1);
        int len=strlen(s+1);
        int opt1=1,opt2=len;
        l=0;
        while(opt1<=opt2)
        {
            if(s[opt1]==s[opt2])
            {
                opt1++;opt2--;
            }
            else break;
        }
        if(opt1>opt2)
        {
            printf("%s\n",s+1);
            continue;
        }
        int ans1=Manacher(opt1,opt2);
        reverse(s+opt1,s+1+opt2);
        l=0;
        int ans2=Manacher(opt1,opt2);
//        cout<<opt1<<' '<<opt2<<' '<<ans1<<' '<<ans2<<endl;
        if(ans1>ans2)
        {
            reverse(s+opt1,s+1+opt2);
            for(int i=1;i<=opt1-1;i++)cout<<s[i];
            for(int i=opt1;i<=opt1+ans1-1;i++)cout<<s[i];
            for(int i=opt1-1;i>=1;i--)cout<<s[i];
            cout<<endl;
        }
        else 
        {
            reverse(s+opt1,s+1+opt2);
            for(int i=1;i<=opt1-1;i++)cout<<s[i];
            for(int i=opt2-ans2+1;i<=len;i++)cout<<s[i];
            cout<<endl;
        }
    }
    return 0;
}

ps：说明一下reverse函数的用法：reverse(s+opt1,s+opt2),s为进行翻转的字符串，opt1为s[opt1]中的opt1，从s[0]开始。如果读入为s+1，也依然是从s[0]开始。opt2为s[0]开始的第opt2个。如果读入为s+1，也依然是s[0]开始的第opt2个。
