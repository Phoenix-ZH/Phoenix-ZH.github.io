A:题目名称：

Contest for Robots

题目大意：

给出长度为1≤n≤100 的两个序列 r,b，且满足 ri, bi∈{0,1}，你需要确定 pi（pi≥1）。

满足
$$
\sum_{i=1}^n r_ip_i > \sum_{i=1}^n b_ip_i
$$
，并且最小化max pi（1<=i<=n）

如果不可能，输出 -1 。

题解：

因为序列的数字要么为1，要么为0，那么如果r中没有数字比b中的大，那么显然不可能实现题目的要求。一旦求得r比b大的个数w1，和r比b小的个数w2，就可以求maxpi，但是注意要求是大于，所以ans=(w2+1)/w1;并且要判断是否整除，如果没有则ans++。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=1000+5;
int a[maxn],b[maxn],n;
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
    for(int i=1;i<=n;i++)scanf("%d",&b[i]);
    int w1=0,w2=0;
    for(int i=1;i<=n;i++)
    {
        if(a[i]>b[i])w1++;
        if(a[i]<b[i])w2++;
    }
    if(w1==0&&w2==0)printf("-1\n");
    else if(w1==0&&w2>0)printf("-1\n");
    else
    {
        int s=(w2+1)/w1;
        if(s*w1!=w2+1)s++;
        printf("%d",s);
    }
    return 0;
}

B：题目名称：

Journey Planning

题目大意：

给出一个长度为 n的序列 b。

设 b的一个子序列 c 的长度为 m，它的第 i个元素在原序列的下标为ci，要求对于任意 1≤i<m，ci+1−ci=bci+1−bci。

此时这个子序列的贡献为
$$
\sum_{i=1}^m{b_{c_i}}
​
$$
求出所有合法的子序列的贡献中最大的贡献。

题解：

有题目大意可知，要求一个子序列，假设该数字在原数组第i位，则要满足bi-bj=i-j;即
bi-i=bj-j;所以只需要sort排序的条件设为x.v-x.id<y.v-x.id；然后线性求和并取max，注意最后的小细节。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const long long maxn=200000+5;
long long n;
struct node
{
    long long v,id;
}a[maxn];
long long cmp(node x,node y)
{
    return x.v-x.id<y.v-y.id;
}
int main()
{
    scanf("%lld",&n);
    for(long long i=1;i<=n;i++)
    {
        scanf("%lld",&a[i].v);
        a[i].id=i;
    }
    sort(a+1,a+1+n,cmp);
    long long maxx=a[1].v,ans=a[1].v;
    for(int i=2;i<=n;i++)
    {
        if(a[i].v-a[i].id==a[i-1].v-a[i-1].id)
        {
            ans+=a[i].v;
            maxx=max(maxx,ans);
        }
        else 
        {
            maxx=max(maxx,ans);
            ans=a[i].v;
        }
    }
    maxx=max(maxx,ans);
    printf("%lld\n",maxx);
    return 0;
}
