A.题目名称：

Captain Flint and Crew Recruitment

题目大意：

如果一个数 x能够表示成两个不同质数的乘积，那我们就称 x是 nearly prime 。给出n，求n能否分成4个不同正整数的和，其中至少三个是 nearly prime

思路：

首先可以取得最小的三个 nearly prime：6，10，14。如果n<=30那么肯定无法得到符合要求的组合。如果n>=45则至少都有6，10，14，所以一定可以得到符合要求的组合。问题就是30<n<45之间的情况，只需要单独分析x=6/10/14（x=n-30）的情况。

代码：

#include<bits/stdc++.h>
using namespace std;
int T,n;
int main()
{
    cin>>T;
    while(T--){
        cin>>n;
        if(n<=30)
        {
            printf("NO\n");
            continue;
        }
        if(n>=45)
        {
            printf("YES\n");
            printf("6 10 14 %d\n",n-30);
            continue;
        }
        else 
        {
            int a=6,b=10,c=14,x=n-30;
            if(x==6)
            {
                a=5,b=6,c=10,x=15;
            }
            if(x==10)
            {
                c=21;x=3;
            }
            if(x==14)
            {
                c=21;x=7;
            }
            printf("YES\n");
            printf("%d %d %d %d\n",a,b,c,x);
        }
    }
    return 0;
}

B.题目名称：

Captain Flint and a Long Voyage

题目大意：

给一个n，求得一个n位的十进制数x，将x转换成二进制数，并且去掉最后n位，要求这个去掉n位的二进制的数最大，问x的最小可能值。

思路：

所有数字都为9，4个二进制位为一组，若一组中全为0则十进制写为8，其余正常转换。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=500000+1000;
int T,n,a[maxn];
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>n;
        int tot=0;
        for(int i=1;i<=n;i++)
        {
            a[++tot]=1;a[++tot]=0;a[++tot]=0;a[++tot]=1;
        }
        for(int i=tot;i>=tot-n+1;i--)
        {
            a[i]=0;
        }
        for(int i=1;i<=tot;i+=4)
        {
            int x=a[i];
            x*=2;x+=a[i+1];
            x*=2;x+=a[i+2];
            x*=2;x+=a[i+3];
            if(x==0)x=8;
            cout<<x;
        }
        cout<<endl;
    }
}

C.题目名称：

Uncle Bogdan and Country Happiness

题目大意：

给出一棵树，每个节点给出最终的人数、最终的心情值。每个人从1号节点出发，在路上可能有些人心情会从good变成bad。每个人沿着最短路径回到自己的节点，求每个节点给出的心情值是否可能都是准确的。（心情值=good人数-bad人数）

思路：

先dfs求得每一个结点经过的总的人数。再dfs求每一个结点所有经过这个结点的人中good的人数和bad的人数。即：good[]+bad[]=p[]，good[]-bad[]=h[]。所以good[]=（p[]+h[])/2,bad[]=(p[]-good[]).如果一个结点的good[]+bad[]!=p[]||good[]-bad[]!=h[]||good[]<0|bad[]<0那么这个结点的心情值就是错误的。然后更新子节点，注意要统计子节点的总的bad的人数。如果当前结点的bad[]-t[] (最终节点的人数)>tot(子节点的bad总数)说明这个组合是错的。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=1e5+50;
struct node
{
    int to,next;
}edge[maxn*2];
int k=1,head[maxn];
void add(int u,int v)
{
    edge[++k].to=v;edge[k].next=head[u];head[u]=k;
}
int p[maxn],h[maxn],p1[maxn],p2[maxn],T,t[maxn];
void dfs1(int u,int f)
{
    for(int i=head[u];i;i=edge[i].next)
    {
        if(edge[i].to==f)continue;
        dfs1(edge[i].to,u);
        p[u]+=p[edge[i].to];
    }
}
int sign=0;
void dfs2(int u,int f)
{
    p1[u]=(p[u]+h[u])/2,p2[u]=p[u]-p1[u];
    if(p1[u]+p2[u]!=p[u]||p1[u]-p2[u]!=h[u]||p1[u]<0||p2[u]<0)sign=1;
    int tot=0,son=0;
    for(int i=head[u];i;i=edge[i].next)
    {
        if(edge[i].to==f)continue;
        dfs2(edge[i].to,u);
        tot+=p2[edge[i].to];
        son=1;
    }
    if(p2[u]-t[u]>tot&&son)sign=1;
}
int main()
{
    cin>>T;
    while(T--)
    {
        int n,m;
        cin>>n>>m;
        sign=0;k=1;memset(head,0,sizeof(head));
        for(int i=1;i<=n;i++)cin>>p[i],t[i]=p[i];
        for(int i=1;i<=n;i++)cin>>h[i];
        for(int i=1;i<n;i++)
        {
            int u,v;
            cin>>u>>v;
            add(u,v);add(v,u);
        }
        dfs1(1,1);
        dfs2(1,1);
        if(sign)printf("NO\n");
        else printf("YES\n");
    }
}

D.题目名称：

Captain Flint and Treasure

题目大意：

给出长度为n的a数组和b数组，一共操作n次（每个位置必须操作一次），若第i次选定位置i，则ans+=a[i]，并且如果b[i]!=-1 a[b[i]]+=a[i]，确定一个操作顺序使得取出的ans最大，并输出操作顺序。

思路：

对于b[i]!=-1的则可以i—->b[i]引一条边。然后进行拓扑排序，如果当前节点的值大于0，则加到下一个节点中，否则就不加入（加了反而使得答案更小），最后按照拓扑排序的顺序输出。（节点值<0的倒序输出防止叠加）

代码：

#include<bits/stdc++.h> 
using namespace std;
const int maxn=2e5+50;
struct node
{
    long long to,next;
}edge[maxn];
long long k=1,head[maxn];
void add(long long u,long long v)
{
    edge[++k].to=v;edge[k].next=head[u];head[u]=k;
}
long long n,a[maxn],b[maxn],rd[maxn],f[maxn],opt,ans,dp[maxn];
void Topsort()
{
    queue<long long>q;
    for(long long i=1;i<=n;i++)
    dp[i]=a[i];
    for(long long i=1;i<=n;i++)
        if(!rd[i])
            q.push(i);
    while(!q.empty())
    {
        long long x=q.front();q.pop();
        f[++opt]=x;ans+=dp[x];
        for(long long i=head[x];i;i=edge[i].next)
        {
            rd[edge[i].to]--;
            dp[edge[i].to]+=max(0ll,dp[x]);
            if(!rd[edge[i].to])
            q.push(edge[i].to);
        }
    }
}
int main()
{
    cin>>n;
    for(long long i=1;i<=n;i++)cin>>a[i];
    for(long long i=1;i<=n;i++)cin>>b[i];
    for(long long i=1;i<=n;i++)
    {
        if(b[i]==-1)continue;
        add(i,b[i]);
        rd[b[i]]++;
    }
    opt=0;
    Topsort();
    cout<<ans<<endl;
    for(long long i=1;i<=n;i++)if(dp[f[i]]>0)cout<<f[i]<<' ';
    for(long long i=n;i>=1;i--)if(dp[f[i]]<=0)cout<<f[i]<<' ';cout<<endl;
}
