A.题目名称

String Similarity

题目大意：

定义两个字符串相似：两个字符串中至少有一个位置的值相同。

现在给出一个长度为2n−1的字符串s，现在让你构造一个长度为n的字符串，使得这个字符串与s的所有长度为n的子串都相似。

思路：

直接将s[n]输出n次

ps：一开始想复杂了，认为cnt0的个数>=n就全输出0，否则全输出1.实际上全输出s[n]就可以了。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=1e3+50;
int T,n;char s[maxn];
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>n;
        cin>>s+1;
        int opt0=0;
        for(int i=1;i<=2*n-1;i++)
            if(s[i]=='0')
                opt0++;
        if(opt0>=n)
        {
            for(int i=1;i<=n;i++)cout<<'0';cout<<endl;
        }
        else 
        {
            for(int i=1;i<=n;i++)cout<<'1';cout<<endl;
        }
    }
    return 0;
}

B.题目名称

RPG Protagonist

题目大意：

有两个人，一个人可以带p单位的重量，另一个人可以带f单位的重量，现在有两种物品，给出两种物品单件的重量以及两种物品的数量，求这两个人最多一共带走多少物品（数量最多）。

思路：

首先保证第一个物品的重量比第二个更小。然后枚举第一个人选择第一个物品的数量（从0到cnt1），然后计算第一个人选择第二个物品的数量（注意判断条件一定要小于等于cnt2），然后贪心求得第二个人选择第一个物品的数量和第二个物品的数量。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=2e5+50;
long long T,n,w1,w2,cnt1,cnt2,p,f;
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>p>>f>>cnt1>>cnt2>>w1>>w2;
        if(w1>w2)
        {
            swap(w1,w2);swap(cnt1,cnt2);
        }
        long long ans=0;
        for(long long i=cnt1;i>=0;i--)
        {
            if(p<i*w1)continue;
            long long ned1=(p-i*w1)/w2;
            long long tot1=i+min(ned1,cnt2);
            long long res1=cnt1-i,tot2=0,res2=cnt2-min(ned1,cnt2);
            if(f<=res1*w1)
            {
                tot2=f/w1;
            }
            else 
            {
                long long ned2=(f-res1*w1)/w2;
                tot2=res1+min(ned2,res2);
            }
            ans=max(tot1+tot2,ans);
        }
        cout<<ans<<endl;
    }
    return 0;
}

C.题目名称

Binary String Reconstruction

题目大意：

对于一个字符串s，通过以下规则构建字符串t。对于一个位置i，如果s[i-x]存在且s[i-x]=1或者s[i+x]存在且s[i+x]=1，那么t[i]=1,否则t[i]=1.现在给出串t，求串s。

思路：

从头开始，判断t[1]是0还是1，还确定s[i+x]，并且vis[i+x]标记为1，然后从2开始线性遍历。

如果t[i]==1：

​ if s[i-x]存在且为1则可以跳过；

​ else if s[i-x]存在但未遍历，则s[i-x]=1,标记一下；

​ else if s[i+x]存在但未遍历，则s[i+x]=1，标记一下；

​ else 无答案存在；

如果t[i]==0:

​ if s[i-x]存在 则它必须为0；

​ if s[i+x]存在 则它必须为0；

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=1e5+50;
int T,x,vis[maxn];char s[maxn],a[maxn];
int main()
{
    cin>>T;
    while(T--)
    {
        memset(vis,0,sizeof(vis));
        cin>>s+1;
        int n=strlen(s+1);
        cin>>x;
        if(s[1]=='1')a[1+x]='1';
        else a[1+x]='0';
        vis[1+x]=1;
        int sign=0;
        for(int i=2;i<=n;i++)
        {
            if(s[i]=='1')
            {
                if(i-x>=1&&vis[i-x]&&a[i-x]=='1')continue;
                else if(i-x>=1&&!vis[i-x])
                {
                    vis[i-x]=1;
                    a[i-x]='1';
                }
                else if(i+x<=n&&!vis[i+x]) 
                {
                    vis[i+x]=1;
                    a[i+x]='1';
                }
                else 
                {
                    sign=1;
                    break;
                }
            }
            else 
            {
                if(i+x<=n&&!vis[i+x])
                {
                    vis[i+x]=1;
                    a[i+x]='0';
                }
                if(i-x>=1&&!vis[i-x])
                {
                    vis[i-x]=1;
                    a[i-x]='0';
                }
                if((i+x<=n&&a[i+x]=='1')||(i-x>=1&&a[i-x]=='1'))
                {
                    sign=1;
                    break;
                }
            }
        }
        for(int i=1;i<=n;i++)if(!vis[i])a[i]='0';
        if(sign)
        {
            printf("-1\n");
        }
        else 
        {
            for(int i=1;i<=n;i++)cout<<a[i];cout<<endl;
        }
    }
    return 0;
}

D.题目名称：

Zigzags

题目大意：

给出一个数组，求满足以下条件的四元组（i，j，k，l）的个数。1<=i<j<k<l<=n ai=ak&&aj=al

思路：

先枚举i与l的位置，则j和k一定在i和l之间，然后求i到l之间与i相同的数目res1和与l相同的数目res2，ans+=res1 res2，注意如果ai==al则可能存在重复，需要去重ans-=res1。然后发现这种情况下，存在j与k顺序相反的情况。需要枚举i和j，让k在i和j之间，l在j之后，然后求得i与j之间和ai相等的总数res1，j之后与aj相等的数目res2，ans-=res1 res2

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=3000+50;
int T,n,a[maxn],sum[maxn][maxn];
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>n;
        for(int i=1;i<=n;i++)cin>>a[i];
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=n;j++)
            {
                if(a[i]==j)
                {
                    sum[i][j]=sum[i-1][j]+1;
                }
                else 
                {
                    sum[i][j]=sum[i-1][j];
                }
            }
        }
        long long ans=0;
        for(int i=1;i<=n;i++)
        {
            for(int l=i+3;l<=n;l++)
            {
                long long res1=sum[l-1][a[l]]-sum[i][a[l]];
                long long res2=sum[l-1][a[i]]-sum[i][a[i]];
                ans+=res1*res2;
                if(a[i]==a[l])
                ans-=res1;
            }
        }
        for(int i=1;i<=n;i++)
        {
            for(int j=i+2;j<=n-1;j++)
            {
                long long res1=sum[j-1][a[i]]-sum[i][a[i]];
                long long res2=sum[n][a[j]]-sum[j][a[j]];
                ans-=res1*res2;
            //    if(a[i]==a[j])
            //    ans+=res1;
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
