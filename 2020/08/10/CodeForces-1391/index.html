A.题目名称：

Suborrays

题目大意：

求一个长度为n的序列，保证任意区间[i,j]之间a[i]|a[i+1]|…|a[j]>=i-j+1.

思路：

直接按序输出1 - n一定是符合的。

代码：

#include<bits/stdc++.h>
using namespace std;
int T,n;
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>n;
        for(int i=1;i<=n;i++)cout<<i<<' ';cout<<endl;
    }
    return 0;
}

B.题目名称：

Fix You

题目大意：

一个(n,m)的矩阵，每一个单位格上有‘D’，‘R’，‘D’：向下移动一格，‘R’：向右移动一格。对于任意位置上的物品需要保证它可以运动到(n,m)，一旦出界就失败了。问至少要改变几个单位格？

思路：

真正有影响的是最后一行和最后一列。若最后一行还有‘D’一定不符合条件，必须改成‘R’；若最后一列还有‘R’一定不符合条件，必须改成‘D’。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=100+50;
int T,n,m;char s[maxn][maxn];
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>n>>m;
        for(int i=1;i<=n;i++)cin>>s[i]+1;
        int ans=0;
        for(int i=1;i<=m;i++)
            if(s[n][i]=='D')ans++;
        for(int i=1;i<=n;i++)
            if(s[i][m]=='R')ans++;
        cout<<ans<<endl;
    }
    return 0;
}

C.题目名称：

Cyclic Permutations

题目大意：

长度为n的序列中的数字各不相同。
对于每一个i，找到一个最大的j(j<i)使得pj>pi，i与j之间连一条无向边；
对于每一个i，找到一个最小的(j>i)使得pj>pi，i与j之间连一条无向边。
问当一个序列n存在环的时候，一共有多少种排列组合。

思路：

根据所给的例子知：只要存在一个i，若左边依次递增，右边依次递减，则不符合要求。可以考虑全排列组合的总数-不符合条件的总数。现在求不符合要求的：最大的数字n有n个可选择位置，总数之之和为
$$
C_n^0+C_n^1+C_n^2+….+C_n^{n-1}
$$
实际上就是2^(n-1)。答案就是：n！-2^(n-1)

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=1e6+50;
const int Mod=1e9+7;
int n;long long p[maxn];
int main()
{
    cin>>n;
    long long ans=1;
    for(long long i=1;i<=n;i++)    
    {
        ans=ans*i%Mod;
    }
    long long ans1=1;
    for(int i=1;i<=n-1;i++)ans1=(ans1*2)%Mod;
    ans=(ans-ans1+Mod)%Mod;
    cout<<ans<<endl;
    return 0;
}

D.题目名称：

505

题目大意：

对于一个二进制矩阵，使得它是一个 good matrix （任意的偶数边的正方形内有奇数个1）.问使得它成为good matrix需要该边多少次？如果无法称为good matrix则输出-1

思路：

保证n<=m.若n>=4一定不符合，4 * 4的方阵包含4个2 * 2，4个奇数相加一定是偶数。所以不符合。n==1的一定是0，即始终符合。现在主要讨论n==2，n==3的情况。

n==2：看每一列的1个数%2，统计奇偶性与列下标不对应的总数，答案就是min（ans，m-ans）

n==3：dp[l][i][j][k]表示第l列的三行数字依次为k、j、i的时候需要改变的总数。先初始化为1e9，第一列的每种情况进行列举。然后从第二列开始枚举，依次枚举当前列的三行数字，和上一列的三行数字，并且保证任意2*2的都是奇数个1（即((j+k+jj+kk)&1)&&((i+ii+j+jj)&1)），得到当前列数字为i，j，k需要改变的次数，更新方程。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=1e6+50;
int n,m;int dp[maxn][2][2][2],a[4][maxn];
int main()
{
    cin>>n>>m;
    if(n>m)swap(n,m);
    if(n>=4)
    {
        printf("-1\n");
        return 0;
    }
    if(n==1)
    {
        printf("0\n");
        return 0;
    }
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=m;j++)
        {
            char c;cin>>c;
            a[i][j]=c-'0';
        }
    }
    if(n==2)
    {
        int ans=0;
        for(int i=1;i<=m;i++)
        {
            int tot=0;
            if(a[1][i]==1)tot++;
            if(a[2][i]==1)tot++;
            tot%=2;
            if(tot!=i%2)ans++;
        }
        cout<<min(ans,m-ans)<<endl;
    }
    if(n==3)
    {
        for(int i=0;i<=1;i++)
        {
            for(int j=0;j<=1;j++)
            {
                for(int k=0;k<=1;k++)
                {
                    dp[1][i][j][k]=3-(a[3][1]==i)-(a[2][1]==j)-(a[1][1]==k);
                    for(int l=2;l<=m;l++)dp[l][i][j][k]=1e9;
                }
            }
        }
        for(int l=2;l<=m;l++)
        {
            for(int i=0;i<=1;i++)
            {
                for(int j=0;j<=1;j++)
                {
                    for(int k=0;k<=1;k++)
                    {
                        for(int ii=0;ii<=1;ii++)
                        {
                            for(int jj=0;jj<=1;jj++)
                            {
                                for(int kk=0;kk<=1;kk++)
                                {
                                    if(((j+k+jj+kk)&1)&&((i+ii+j+jj)&1))
                                    {
                                        int ans=3-(a[3][l]==i)-(a[2][l]==j)-(a[1][l]==k);
                                        dp[l][i][j][k]=min(dp[l][i][j][k],dp[l-1][ii][jj][kk]+ans);
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        int ans=1e9;
        for(int i=0;i<=1;i++)
        {
            for(int j=0;j<=1;j++)
            {
                for(int k=0;k<=1;k++)
                {
                    ans=min(ans,dp[m][i][j][k]);
                }
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
