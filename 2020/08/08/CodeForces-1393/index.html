A.题目名称

Rainbow Dash, Fluttershy and Chess Coloring

题目大意：

给定一个正方形区域上色，要求上色的块相邻之间不能相同，也就是我们会在每回合上不同的颜色，要求上色的块邻边是边界或是上一个以上色的块。求至少要多少个回合才能把这个正方形区域完。

思路：

第一个回合，我们只能在边界的块上上色，且 要间隔开一格子上色，那么我们至少可以把边界的一半给上完，那么在下个回合，我们用不同的颜色上色自然可以把最外面的边界给上完。依次这样， 所以我们还是按着之前的块相邻来填，我们发现，进行完这个回合我们的区域和上个回合的是一样的，只不过长度减了2，我们每次进行这种回合都会这样子，那么我们对边长n/2即可，最后再加1.

代码：

#include<bits/stdc++.h>
using namespace std;
long long T,n;
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>n;
        long long ans=n/2+1;
        cout<<ans<<endl;
    }
    return 0;
}

B.题目名称

Applejack and Storages

题目大意：

小马们想做一个方形的和矩形的马厩，每个马厩需要四块木板。现在家具场有一些木板，每次回增加或者减少一个木板。问每次增加或者减少的时候能不能符合要求早出一个方形和一个矩形的马厩。

比赛思路：

tot[]记录a[i]的个数，cnt[]记录tot[x]的数量。在q个查询中，依次更新tot[],cnt[]。如果cnt[2]和cnt[3]的数量>=2&&cnt[4],cnt[5],cnt[6],cnt[7]任意存在一个，则符合条件。如果cnt[2],cnt[3]任意存在一个&&cnt[6],cnt[7]任意存在一个，则符合条件。如果cnt[4]+cnt[5]+cnt[6]+cnt[7]的数量>=2则符合条件，如果存在一个tot[x]>=8则符合条件。其余都不符合。

比赛代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=2e5+10;
int a[maxn],n,tot[maxn],q,cnt[maxn],t,vis[maxn];
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)cin>>a[i],tot[a[i]]++;
    for(int i=1;i<=1e5;i++)
    {
        if(tot[i]==0)continue;
        cnt[tot[i]]++;
        if(tot[i]>=8)t++;
    }
    cin>>q;
    for(int i=1;i<=q;i++)
    {
        char c;int x;
        cin>>c>>x;
        if(c=='+')
        {
            cnt[tot[x]]--;
            tot[x]++;
            cnt[tot[x]]++;
            if(tot[x]==8)t++;
        }
        else 
        {
            if(tot[x]==8)t--;
            cnt[tot[x]]--;
            tot[x]--;
            cnt[tot[x]]++;
        }
        if((cnt[2]+cnt[3]>=2)&&(cnt[4]||cnt[5]||cnt[6]||cnt[7]))
        {
            printf("YES\n");
            continue;
        }
        if((cnt[2]||cnt[3])&&(cnt[6]||cnt[7]))
        {
            printf("YES\n");
            continue;
        }
        if(cnt[4]+cnt[5]+cnt[6]+cnt[7]>=2)
        {
            printf("YES\n");
            continue;
        }
        if(t)
        {
            printf("YES\n");
            continue;
        }
        printf("NO\n");
        continue;
    }
    return 0;
}

优化思路：

维护一个fangnum和一个junum，然后当木板增加减少到了边界的时候，就进行一次判断，更新fangnum和junum的数量，最后通过fangnum和junum判断是否能够成功搭建。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=1e5+50;
int n,a[maxn],q,fangnum,junum,tot[maxn];
int main()
{
    cin>>n;
    int maxx=0;
    for(int i=1;i<=n;i++)cin>>a[i],maxx=max(maxx,a[i]),tot[a[i]]++;
    for(int i=1;i<=maxx;i++)
    {
        fangnum+=tot[i]/4;
        junum+=tot[i]/2;
    }
    cin>>q;
    for(int i=1;i<=q;i++)
    {
        char c;int x;
        cin>>c>>x;
        if(c=='+')
        {
            tot[x]++;
            if(tot[x]%4==2)
            {
                junum++;
            }
            if(tot[x]%4==0)
            {
                fangnum++;
                junum++;
            }
        }
        else 
        {
            if(tot[x]%4==2)
            {
                junum--;
            }
            if(tot[x]%4==0)
            {
                fangnum--;
                junum--;
            }
            tot[x]--;
        }
        if(fangnum>=2||(fangnum>=1&&(junum-fangnum*2)>=2))
        {
            printf("YES\n");
        }
        else 
        {
            printf("NO\n");
        }
    }
    return 0;
}

C.题目名称

Pinkie Pie Eats Patty-cakes

题目大意：

给你一些数字，让你把这些数字尽可能的分散开来，问两个相同数字的最小间距的最大值是多少。

思路：

用tot[]统计a[i]的数量，并且将a[i]重新排序，相同的排一起，数量多的排在前面。然后就是二分，判断条件就是x=x+mid+1，x>n就不符合。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=1e5+10;
int a[maxn],T,n,vis[maxn];
struct node
{
    int t,id;
}tot[maxn];
int maxx=0;
bool check(int mid)
{
    for(int i=1;i<=maxx;i++)
    {
        if(tot[i].t==0)break;
        int x=i;
        for(int j=2;j<=tot[i].t;j++)
        {
            x=x+mid+1;
            if(x>n)return 0;
        }
    }
    return 1;
}
int cmp(node x,node y)
{
    return x.t>y.t;
}
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>n;
        maxx=0;
        memset(tot,0,sizeof(tot));
        for(int i=1;i<=n;i++)cin>>a[i],tot[a[i]].t++,tot[a[i]].id=a[i],maxx=max(maxx,a[i]);
        sort(tot+1,tot+1+maxx,cmp);
        int now=0;
        for(int i=1;i<=maxx;i++)
        {
            if(tot[i].t==0)break;
            for(int j=1;j<=tot[i].t;j++)
            {
                a[++now]=tot[i].id;
            }
        }
        int l=0,r=n,ans=0;
        while(l<=r)
        {
            int mid=(l+r)/2;
            if(check(mid))
            {
                l=mid+1;
                ans=mid;
            }
            else 
            {
                r=mid-1;
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}

D.题目名称

Rarity and New Dress

题目大意：

小红有强迫症，她必须要方形的裙子，但是必须是旋转45度的那种方形，并且裙子的颜色必须一样。

思路：

对于每一个位置我们设dp[i][j]为位置[i][j]向上能最多”延申“的”斜正方形边长“，那么总的答案就是所有位置的dp值之和。转移分为当前位置[i][j]与上面四个位置（[i-1][j]、[i-1][j-1]、[i-1][j+1]、[i-2][j]）能否形成“斜正方形”： 若能，则取min（dp[i-1][j-1]、dp[i-1][j+1]、dp[i-2][j]）+1，这里可以简单画个图理解一下当前[i][j]高度的“延申”只是依赖于这三个位置的dp值， 若不能则dp[i][j]=1.

代码：

#include<bits/stdc++.h>
using namespace std;
const long long maxn=2000+50;
long long n,m,dp[maxn][maxn];char a[maxn][maxn];
int main()
{
    cin>>n>>m;
    for(long long i=1;i<=n;i++)cin>>a[i]+1;
    long long ans=0;
    for(long long i=1;i<=n;i++)
    {
        for(long long j=1;j<=m;j++)
        {
            if(i-1<1||i-2<1||j-1<1|j+1>m||a[i-1][j]!=a[i][j]||a[i-1][j-1]!=a[i][j]||a[i-1][j+1]!=a[i][j]||a[i-2][j]!=a[i][j])
                dp[i][j]=1;
            else 
                dp[i][j]=min(min(dp[i-1][j-1],dp[i-1][j]),min(dp[i-1][j+1],dp[i-2][j]))+1;
            ans+=dp[i][j];
        }
    }
    cout<<ans<<endl;
    return 0;
}
