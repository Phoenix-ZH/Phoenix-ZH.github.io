A.题目名称

Omkar and Password

题目大意：

将任意两个相邻的大小不同的数字合并，问数组长度最小可以为多少？

思路：

每次将最大的和别的合并。如果所有都相同，长度为n；否则长度为1.

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=2e5+60;
int T,n,a[maxn];
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>n;
        for(int i=1;i<=n;i++)cin>>a[i];
        int sign=0;
        for(int i=2;i<=n;i++)
        {
            if(a[i]!=a[i-1])
            {
                sign=1;
                break;
            }
        }
        if(sign)printf("1\n");
        else printf("%d\n",n);
    }
    return 0;
}

B.题目名称

Omkar and Infinity Clock

题目大意：

在一个长度为n的数组中操作k次。每次操作选择最大的数字ai为d，然后用d-ai（1<=i<=n）。问k次操作后数组的数字。

思路：

首先操作一次，记录数组，注意每次取maxx的初始值为-1e18，因为-1e9<=ai<=1e9，然后判断k-1的奇偶性，如果为偶就是该数组，否则再取maxx，然后maxx-ai，就是所求数组。

代码：

#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const int maxn=2e5+50;
LL n,T,a[maxn],k;
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>n>>k;
        for(int i=1;i<=n;i++)cin>>a[i];
        LL maxx=-1e18;
        for(int i=1;i<=n;i++)maxx=max(maxx,a[i]);
        k--;
        for(int i=1;i<=n;i++)a[i]=maxx-a[i];
        if(k%2==0)
        {
            for(int i=1;i<=n;i++)cout<<a[i]<<' ';cout<<endl;
        }
        else 
        {
            maxx=-1e18;
            for(int i=1;i<=n;i++)maxx=max(maxx,a[i]);
            for(int i=1;i<=n;i++)a[i]=maxx-a[i];
            for(int i=1;i<=n;i++)cout<<a[i]<<' ';cout<<endl;
        }
    }
    return 0;
}

C.题目名称

Omkar and Waterslide

题目大意：

将一串数字变成非递减，每次操作可以将一串连续的非递减数列都+1，问最少需要多少次操作？

思路：

对于a[i]<=a[i+1], 不需要对a[i+1]进行操作；

对于a[i]>a[i+1],需要将a[i+1]加到a[i]，即ans+=a[i+1]-a[i];

现在把a[i+1]加到和a[i]一样大，a[i+1]就可以和a[i]一起操作，所以当a[i]>=a[i-1]的时候，a[i+1]也会跟着加。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=2e5+50;
long long T,n,a[maxn];
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>n;
        long long ans=0;
        for(int i=1;i<=n;i++)cin>>a[i];
        for(int i=1;i<n;i++)
        ans+=max(0ll,a[i]-a[i+1]);
        cout<<ans<<endl;
    }
}

思路-2：

这是比赛时候的时候，写得特别麻烦，对于a[i]>a[i+1]的直接将ans+=a[i]-a[i+1],记录一串不符合条件（非递减）的数列的最大值maxx，一旦遇到a[i]>a[i-1],ans+=a[i]-maxx，因为之前的已经非递减了，只要加到a[i]就好了，最后这一串数列结束，ans+=a[opt-1]-maxx，即加到刚好非递减

代码：

#include<bits/stdc++.h>
using namespace std;
const long long maxn=2e5+50;
long long T,n,a[maxn];
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>n;
        for(long long i=1;i<=n;i++)cin>>a[i];
        long long opt=2,ans=0;
        while(opt<=n)
        {
            if(a[opt-1]>a[opt])
            {
                long long now=opt,maxx=a[opt],minn=a[opt];
                if(a[now+1]>a[opt-1])
                {
                    ans+=a[opt-1]-a[opt];
                    opt=now+1;
                }
                else 
                {
                    now=opt+1;
                    while(now<=n&&a[now]<a[opt-1])
                    {
                        if(a[now-1]>a[now])
                        {
                            ans+=a[now-1]-a[now];
                        }
                        else if(a[now]>a[now-1])
                        {
                            if(a[now]>maxx)
                            {
                                ans+=a[now]-maxx;
                                maxx=a[now];
                            }
                        }
                        now++;
                    }
                    ans+=a[opt-1]-maxx;
                    opt=now;
                }
            }
            else opt++;
        }
        cout<<ans<<endl;
    }
    return 0;
}

D.题目名称

Omkar and Bed Wars

题目大意：

一个人同时被左右两两边的人攻击，或者左右两边的人都没攻击他，那他可随便攻击一个人。一个人若只被一个人攻击，他就要攻击回去。问如何该边人的攻击，使得用最少的改变次数，使得攻击符合条件。

思路：

当a[j-1]=R,a[j]=R,a[j+1]=R的时候对于a[j]的攻击就不符合条件；当a[j-1]=L,a[j]=L,a[j+1]=L的时候对于a[j]的攻击也是不符合条件。因此要时候没有>=3的连续L/R出现。

当全是L/R的时候答案就是(n+2)/3;

然后将所有的连续子串加入数组中，判断s[1]与s[n],如果s[1]=s[n]，就要就首尾合并，最后ans+=a[i]/3;

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=2e5+50;
int n,T;char s[maxn];
int main()
{
    cin>>T;
    while(T--)
    {
        vector<int>a;
        cin>>n;
        cin>>s+1;
        int sign=0;
        for(int i=2;i<=n;i++)
        {
            if(s[i-1]!=s[i])
            {
                sign=1;
                break;
            }
        }
        if(sign==0)
        {
            cout<<(n+2)/3<<endl;
            continue;
        }
        sign=1;
        for(int i=2;i<=n+1;i++)
        {
            if(s[i]!=s[i-1])
            {
                a.push_back(sign);
                sign=1;
            }
            else 
                sign++;
        }

        if(s[1]==s[n])
        {
            a[0]+=a[a.size()-1];
            a.pop_back();
        }
        int ans=0;
        for(int i=0;i<a.size();i++)ans+=a[i]/3;
        cout<<ans<<endl;
    }
    return 0;
}
