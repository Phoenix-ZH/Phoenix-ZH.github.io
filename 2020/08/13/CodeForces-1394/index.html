A.题目名称

Boboniu Likes to Color Balls

题目大意：

r个红球，g个绿球，b个蓝球，w个白球。可以任意次将一个红球和一个绿球和一个蓝球都转化成白球。是否可以组成回文色？

思路：

当偶数色的数目>=3则一定可以。若只有一个偶数色，则红绿蓝都至少有一个就可以。其余的都是不可以的。

代码：

#include<bits/stdc++.h>
using namespace std;
long long T,a,b,c,d;
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>a>>b>>c>>d;
        int tot=0;
        if(a%2==0)tot++;
        if(b%2==0)tot++;
        if(c%2==0)tot++;
        if(d%2==0)tot++;
        if(tot>=3||tot==0)
        {
            printf("Yes\n");
        }
        else 
        {
            if(tot==1)
            {
                if(a>=1&&b>=1&&c>=1)printf("Yes\n");
                else printf("No\n");
            }
            else printf("No\n");
        }
    }
    return 0;
}

B.题目名称

Boboniu Plays Chess

题目大意：

n*m的棋盘，初始位置在非边界处，每次可以移动到同一行或者同一列未到过的位置上。请输出符合情况的答案。

思路：

先左平移到边界，在从初始位置的右边开始右平移到边界，然后列下标依次减小，对于行下标一开始是由小到大，再由大到小，交替进行，类似于S线。注意开始的初始位置所在的行不可以经过。

代码：

#include<bits/stdc++.h>
using namespace std;
int n,m,si,sj;
int main()
{
    cin>>n>>m>>si>>sj;
    int j=sj;
    while(sj!=0)
    {
        printf("%d %d\n",si,sj);
        sj--;
    }
    sj=j+1;
    while(sj<=m)
    {
        printf("%d %d\n",si,sj);
        sj++;
    }
    sj=m;
    int now=1;
    for(j=m;j>=1;j--)
    {
        if(now%2==1)
        {
            for(int i=1;i<=n;i++)
            {
                if(i==si)continue;
                printf("%d %d\n",i,j);
            }    
            now++;
        }
        else 
        {
            for(int i=n;i>=1;i--)
            {
                if(i==si)continue;
                printf("%d %d\n",i,j);
            }
            now++;
        }
    }
    return 0;
}

C.题目名称

Boboniu and Bit Operations

题目大意：

a[]（长度为n）,b[]（长度为m），对于每一个ai（1<=i<=n）选择一个bj，ci=ai&bj。求能得到的最小的c1|c2|…|cn。

思路：

观察到0<=ai<2^9,所以最大的答案就是2^10 - 1.直接从小到大枚举答案，能符合的最小的就是所求答案。接下来就是如何进行判定了：对于所枚举的答案，需要让a[]与所有b[]进行操作，一旦存在ci|mid<=mid说明当前ai是符合的，如果不存在就说明枚举的答案是不对的。

代码：

#include<bits/stdc++.h>
using namespace std;
const int maxn=200+5;
int n,m,a[maxn],b[maxn],c[maxn];
bool check(int mid)
{
    memset(c,0,sizeof(c));
    for(int i=1;i<=n;i++)
    {
        int sign=0;
        for(int j=1;j<=m;j++)
        {
            c[i]=a[i]&b[j];
            if((c[i]|mid)<=mid)
            {
                sign=1;
                break;
            }
        }
        if(!sign)return 0;
    }
    return 1;
}
int main()
{
    cin>>n>>m;
    for(int i=1;i<=n;i++)cin>>a[i];
    for(int i=1;i<=m;i++)cin>>b[i];
    sort(a+1,a+1+n);sort(b+1,b+1+m);
    int l=0,r=pow(2,10),ans=0;
    for(int i=0;i<=pow(2,10);i++)
    {
        if(check(i))
        {
            cout<<i<<endl;
            return 0;
        }
    }
}

D.题目名称

Boboniu Chats with Du

题目大意：

Du可以在群里聊天n天，如果Du当前选择ai>m（Boboniu的恒定心情值），则Du会被禁言d天，问最大的ai的和。

思路：

将a[]由小到大排序，可以枚举ai<=m的个数，然后利用前缀和得到ai<=m的和（肯定优先选择大的），剩下的天数就是发一次言就会被禁言，一次发言的耗费就是d+1天，注意求次数时候向上取整，这里也是前缀和。

ps：千万别忘了枚举天数别忘了枚举0天的情况！

代码：

#include<bits/stdc++.h>
using namespace std;
const long long maxn=1e5+50;
long long n,d,m,a[maxn],sum[maxn],sum2[maxn];
int main()
{
    cin>>n>>d>>m;
    for(long long i=1;i<=n;i++)cin>>a[i];
    sort(a+1,a+1+n);
    long long opt=n+1;
    for(long long i=1;i<=n;i++)
    {
        if(a[i]<=m)continue;
        else 
        {
            opt=i;
            break;
        }
    }
    for(long long i=1;i<=opt-1;i++)sum[i]=sum[i-1]+a[i];
    long long t=0;
    for(long long i=n;i>=opt;i--)sum2[++t]=sum2[t-1]+a[i];
    long long ans=0;
    for(long long i=min(n,opt-1);i>=0;i--)
    {
        long long tot=sum[opt-1]-sum[opt-1-i];
        long long res=n-i;
        long long c=ceil(res*1.0/(1+d)*1.0);
        tot+=sum2[c];
        ans=max(ans,tot);
    }
    cout<<ans<<endl;
    return 0;
}
