A: 题目名称：

Two Rabbits

题目大意：

有T组数据，给定x，y，a，b。其中x，y分别为两兔子的坐标(x<y),a，b分别为对应兔子的速度，左边的兔子向右跳，右边的兔子向左跳，问能否相遇，若能请输出时间t，否则输出-1.

题解：

设delt=y-x，若delt%(a+b)则t=delt/(a+b),否则输出-1.

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
long long t,x,y,a,b;
int main()
{
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld%lld%lld",&x,&y,&a,&b);
        long long delt=abs(y-x);
        if(delt%(a+b)==0)
        {
            long long t=delt/(a+b);
            printf("%d\n",t);
            continue;
        }
        else 
        {
            printf("-1\n");
            continue;
        }
    }
    return 0;
}

B：题目名称：

Longest Palindrome

题目大意：

给定n个字符串，每个字符串的长度为m，可以任意选择若干个字符串随机组合构成最长的回文串，如果有多个这样的回文串，任意输出一个即可，如果没有则输出0.

题解：

枚举所有的字符串，找到两个字符串中：字符串i从头到尾等于字符串j的丛尾到头。这样i和j可以配对，并且标记一下。最后还要考虑一种情况：一个字符串本身就是回文，它可以放在中间，并且它没有被标记过，找到一个即可。最后判断是否存在，若无则输出0，否则先输出长度ans * 2 * m+Ans * m，然后先输出前面部分的字符串，并储存方便后面逆序输出，然后输出中间部分，若不存在则不管。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=100+5;
int n,m,vis[maxn],stack[maxn],V[maxn];char s[maxn][maxn],t[maxn*maxn];
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)scanf("%s",s[i]);
    int ans=0,Ans=0;
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=n;j++)
        {
            if(i==j||vis[i]||vis[j])continue;
            int sign=0;
            for(int k=0;k<m;k++)
            {
                if(s[i][k]!=s[j][m-k-1])
                {
                    sign=1;
                    break;
                }
            }
            if(sign==0)
            {
                vis[i]=j;vis[j]=i;
                ans++;
                stack[ans]=i;
            }
        }
    }
    for(int i=1;i<=n;i++)
    {
        if(vis[i])continue;
        int sign=0;
        for(int j=0;j<m;j++)
        {
            if(s[i][j]!=s[i][m-j-1])
            {
                sign=1;
                break;
            }
        }
        if(sign==0)
        {
            Ans++;
            V[Ans]=i;
            break;
        }
    }
    if(ans==0&&Ans==0)
    {
        printf("0\n");
    }
    else 
    {
        printf("%d\n",ans*2*m+Ans*m);
        int now=0;
        for(int i=1;i<=ans;i++)
        {
            printf("%s",s[stack[i]]);
            for(int j=0;j<m;j++)
            {
                t[++now]=s[stack[i]][j];
            }
        }
        if(Ans)
        {
            printf("%s",s[V[Ans]]);
        }
        for(int i=now;i>=1;i--)
        printf("%c",t[i]);
        cout<<endl;
    }
    return  0;
}

C：题目名称：

Air Conditioner

题目大意：

有q组数据，每组数据给定顾客数n，和初始温度m，然后再给定每个顾客到达的时间，和适应的最低温度和最高温度， 每一分钟可以选择升高温度1，降低温度1，或者不变。问能否满足所有的顾客，能则YES，否则NO。

题解：

一开始我以为是贪心，结果准备统计所有的空闲时间并且与调整时间来比大小，结果不对，完全错误，于是又想DP，但是数据10^9，没办法写出DP方程，然后突然想到了区间，既然给定了左右端点，那不就可以得到每个顾客到达那一刻温度可调整的区间吗？只要这个区间和顾客适应的区间存在交点那就符合。并且保证判断之后这个区间要修改到顾客适应的区间的范围内(一开始以为既然符合情况了就不会存在超出范围的情况了，其实完全可能某一端点无法在限定时间内缩回去，所以超出的必须除去)，因为当前温度超出顾客的适应范围就不符合要求了。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=500+5;
long long T,n,t,x,y;
struct node
{
    long long t,x,y;
}a[maxn];
int main()
{
    scanf("%lld",&T);
    while(T--)
    {
        scanf("%lld%lld",&n,&t);
        x=y=t;
        for(int i=1;i<=n;i++)
        scanf("%lld%lld%lld",&a[i].t,&a[i].x,&a[i].y);
        long long now=1;
        long long sign=0;
        while(now<=n)
        {
            long long delt=a[now].t-a[now-1].t;
            if(x<a[now].x)
            {
                x+=delt;
                if(x>a[now].x)
                x=a[now].x;
            }
            else if(x>a[now].x)
            {
                x-=delt;
                if(x<a[now].x)
                x=a[now].x;
            }
            if(y<a[now].y)
            {
                y+=delt;
                if(y>a[now].y)
                y=a[now].y;
            }
            else if(y>a[now].y)
            {
                y-=delt;
                if(y<a[now].y)
                y=a[now].y;
            }
            if((x<a[now].x&&y<a[now].x)||(x>a[now].y&&y>a[now].y))
            {
                sign=1;
                printf("NO\n");
                break;
            }
            if(x<a[now].x)x=a[now].x;
            if(y>a[now].y)y=a[now].y;
            now++;
        }
        if(sign==0)
        {
            printf("YES\n");
        }
    }
    return 0;
}

D：题目名称：

Shortest and Longest LIS

题目大意：

有t组数据，每组数据给定一个n，说明数字1-n，然后给定n-1个>/<，让排出符合要求的数字排列，第一个要求最长上升序列是最短的，第二个要求最长上升序列是最短的。

题解：

贪心+模拟。首先考虑第一问，观察一下答案，发现凡是小于(不连续)的都是从左到右依次减小，而大于的都是左边的尽量大，右边的尽量小。可以设想：要让最长上升序列最短，那么<对应的数字就要比它左边的<对应的数字更小，这样就可以进行隔断，来缩小长度；而左边>对应的数字就是比右边>对应的数字更小，这样的话越右边的数字就越小，这样也可以缩小长度。于是对于问题一的做法就很简单了：设minn=1,maxx=n，因为越右边的<对应的数字越小，所有我们从右向左枚举，一旦枚举到s[i]==’<’就让a[i]=minn并且minn++，但是现在问题来了：如果<重复了呢？(如<<,<<<等等)再如此做的话就与’<’的定义矛盾了，于是这时候就需要找到这一串连续的 ‘<’的左端点，然后从左端点到右端点依次让a[j]=minn并且minn++。枚举结束后让i=now+1，这样就实现了跳转，依旧是线性的。然后对于’>’，要让左边的尽量大，所以我们从左向右枚举，如果s[i]==’>’,那么a[i]=maxx，maxx–；最后a[n]=maxx;这样再输出a数组，问题1就解决了。

那问题2呢？

问题2要求的是最长上升序列最长，就与问题1相反，对于’<’就从左往右依次增大，对于’>’就从右往左依次减小，方法与问题基本一样，但是要注意连续的’>’的情况，处理方式与连续的’<’类似，如果s[n-1]即最后一个符号是 <，那么就让a[n]=n,这里特判一下即可。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=200000+5;
int T,n;char s[maxn];int a[maxn],maxx,minn;
int main()
{
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d",&n);scanf("%s",s+1);
        maxx=n;minn=1;
        for(int i=n-1;i>=1;i--)
        {
            if(s[i]=='<'&&s[i-1]!='<')
            {
                a[i]=minn;
                minn++;
            }
            else if(s[i]=='<'&&s[i-1]=='<')
            {
                int now=i;
                while(s[now]=='<'&&now>=1)now--;
                for(int j=now+1;j<=i;j++)
                {
                    a[j]=minn;
                    minn++;
                }
                i=now+1;
            }
        }
        for(int i=1;i<=n-1;i++)
        {
            if(s[i]=='>')
            {
                a[i]=maxx;
                maxx--;    
            }
        }
        a[n]=maxx;
        for(int i=1;i<=n;i++)
        printf("%d ",a[i]);
        cout<<endl;
        maxx=n,minn=1;
        for(int i=1;i<=n-1;i++)
        {
            if(s[i]=='<')
            {
                a[i]=minn;
                minn++;
            }
        }
        if(s[n-1]=='<')
        {
            a[n]=n;maxx--;
        }
        for(int i=n-1;i>=1;i--)
        {
            if(s[i]=='>'&&s[i-1]!='>')
            {
                a[i]=maxx;
                maxx--;
            }
            else if(s[i]=='>'&&s[i-1]=='>')
            {
                int now=i;
                while(s[now]=='>'&&now>=1)now--;
                for(int j=now+1;j<=i;j++)
                {
                    a[j]=maxx;
                    maxx--;
                }
                i=now+1;
            }
        }
        if(s[n-1]!='<')
        {
            a[n]=maxx;
        }
        for(int i=1;i<=n;i++)
        {
            printf("%d ",a[i]);
        }
        cout<<endl;
    }
    return 0;
}

E：题目名称：

1-Trees and Queries

题目大意：

给定一个n个点的树,相邻点的距离为1 。

q个询问，每个询问包含无个整数： x,y,a,b,k。

含义是：在原树上新连上一条边 (x,y) ，要求判断一下从a点是否有距离为k的到b的路径。

注意：

  • 每个询问是独立的，即上次询问加上的边，不能为这一次的询问所用。
  • 这一条路径也许会重复经过某一条边或某一点。

(q<=10^5,n<=10^5)

题解：

由题知此图是树，任意两点之间的最短距离是确定唯一的。有q个操作，每次操作会加入两种新边，一共就有三种边(对于在无关路径上来回移动的不考虑)，很显然每次新加入边之后，受到直接影响的是端点x，y。它将改变a–>b的路径方式。考虑新的路径必然是将端点a，b作为切入点。(必须要走新加入的边，因为这个边是改变路径方式的唯一方式！)

路径1：a–>b；

路径2：a–>x x–>y(即新加的边) y–>b

路径3：a–>y y–>x(即新加的边) x–>b

那么就开始求路径1，2，3的长度。如何求呢？肯定需要用到LCA了。为了方便，设点1为起点，并且deep[1]=0,假如要求点a，b的距离，那么就要分两种情况：

情况1：a，b其中一点是另一个点的祖先，那么s=abs(deep[a]-deep[b]);

情况2：f1是点a，b的最近公共祖先，那么s=deep[a]+deep[b]-deep[f1]*2(第一次交就因为没×2 WA了)

对于路径1的s：

        f1=lca(x,y);
        if(f1==x||f1==y)
        s1=abs(deep[x]-deep[y]);
        else
        s1=deep[x]+deep[y]-deep[f1]*2;

对于路径2的s：

        f1=lca(x,a);f2=lca(b,y);//因为是两条边，自然两边的LCA都要求
        if(f1==x||f1==a)
        s1=abs(deep[x]-deep[a]);
        else 
        s1=deep[x]+deep[a]-deep[f1]*2;
        if(f2==b||f2==y)
        s1=s1+abs(deep[b]-deep[y]);
        else 
        s1=s1+deep[b]+deep[y]-deep[f2]*2;
        s1+=1;//必须要走新加入的边，因为这个边是改变路径方式的唯一方式！

对于路径3的s：(同路径2类似)

        f1=lca(x,b);f2=lca(a,y);
        if(f1==x||f1==b)
        s1=abs(deep[x]-deep[b]);
        else 
        s1=deep[x]+deep[b]-deep[f1]*2;
        if(f2==a||f2==y)
        s1=s1+abs(deep[a]-deep[y]);
        else 
        s1=s1+deep[a]+deep[y]-deep[f2]*2;
        s1+=1;

最后如何判定s是否符合要求呢？

首先明确一下，如果如果剩余了偶数的长度，其实就可以在终点和它相邻的点之间来回走，直到走完为止。当然要确保k足够提前到达终点。

所以判定条件就是(k-s1)>=0&&(k-s1)%2==0。三条路径任意一条满足即可，否则就NO。

ps:我人傻了，为什么要分情况啊，直接s1=deep[a]+deep[b]-deep[f1]*2就可以了啊;

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const long long maxn=100000+5;
long long K=1,n,q,head[maxn],father[maxn][22],vis[maxn],deep[maxn];
struct node
{
    long long to,next;
}edge[maxn*2];
void add(long long u,long long v)
{
    edge[++K].to=v;edge[K].next=head[u];head[u]=K;
}
long long dfs(long long x,long long depth)
{
    deep[x]=depth;vis[x]=1;
    for(long long i=head[x];i;i=edge[i].next)
    {
        if(!vis[edge[i].to])
        {
            father[edge[i].to][0]=x;
            dfs(edge[i].to,depth+1);
        }
    }
}
long long lca(long long x,long long y)
{
    if(deep[x]<deep[y])swap(x,y);
    long long depth=deep[x]-deep[y];
    for(long long i=20;i>=0;i--)
    {
        if(depth&(1<<i))
        {
            x=father[x][i];
        }
    }
    if(x==y)return x;
    for(long long i=20;i>=0;i--)
    {
        if(father[x][i]!=father[y][i])
        {
            x=father[x][i];y=father[y][i];
        }
    }
    return father[x][0];
}
int main()
{
    scanf("%lld",&n);
    for(long long i=1;i<n;i++)
    {
        long long a,b;
        scanf("%lld%lld",&a,&b);
        add(a,b);add(b,a);
    }
    dfs(1,0);
    for(long long i=1;i<=20;i++)
    {
        for(long long j=1;j<=n;j++)
        {
            father[j][i]=father[father[j][i-1]][i-1];
        }
    }
    scanf("%lld",&q);
    for(long long i=1;i<=q;i++)
    {
        long long a,b,x,y,k,f1,f2,s1;//新加a--b 
        long long sign=0;
        scanf("%lld%lld%lld%lld%lld",&a,&b,&x,&y,&k);
        f1=lca(x,y);
        if(f1==x||f1==y)
        s1=abs(deep[x]-deep[y]);
        else
        s1=deep[x]+deep[y]-deep[f1]*2;
        if((k-s1)>=0&&(k-s1)%2==0)
        {
            sign=1;
            printf("YES\n");
            continue;
        }
        f1=lca(x,a);f2=lca(b,y);
        if(f1==x||f1==a)
        s1=abs(deep[x]-deep[a]);
        else 
        s1=deep[x]+deep[a]-deep[f1]*2;
        if(f2==b||f2==y)
        s1=s1+abs(deep[b]-deep[y]);
        else 
        s1=s1+deep[b]+deep[y]-deep[f2]*2;
        s1+=1;
        if((k-s1)>=0&&(k-s1)%2==0)
        {
            sign=1;
            printf("YES\n");
            continue;
        }
        f1=lca(x,b);f2=lca(a,y);
        if(f1==x||f1==b)
        s1=abs(deep[x]-deep[b]);
        else 
        s1=deep[x]+deep[b]-deep[f1]*2;
        if(f2==a||f2==y)
        s1=s1+abs(deep[a]-deep[y]);
        else 
        s1=s1+deep[a]+deep[y]-deep[f2]*2;
        s1+=1;
        if((k-s1)>=0&&(k-s1)%2==0)
        {
            sign=1;
            printf("YES\n");
            continue;
        }
        if(sign==0)
        {
            printf("NO\n");
        }
    }
    return 0;
}
