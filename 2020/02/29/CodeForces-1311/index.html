A：题目名称：

Add Odd or Subtract Even

题目大意：

  • 给定两个正整数a,b。
  • 每次操作你可以把 a加上某个正奇数，或者减去某个正偶数。
  • 问至少多少次操作可以把 a变成 b。
  • 多组数据 :T<=10^4,1<=a,b<=10^9

题解：

对于a，b关系可分为3种情况：a>b;a==b;a b:若a-b为偶数，答案为1；否则为2，先减一个偶数让a=b-1，再加1。
a < b:若b-a为奇数，答案为1；否则为2.

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int T,a,b;
int main()
{
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d%d",&a,&b);
        if(b>a)
        {
            if((b-a)%2==1)printf("1\n");
            else printf("2\n");
        }
        else if(b==a)
        {
            printf("0\n");
        }
        else if(b<a)
        {
            if((a-b)%2==0)printf("1\n");
            else printf("2\n");
        }
    }
    return 0;
}

B：题目名称：

WeirdSort

题目大意：

题解：

这题看起来和冒泡排序很像，将冒泡排序的判断条件改一下就可以，每一次必须是在限制的位置上时才可以swap。最后再判断是否变成有序数列即可。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=100+5;
int a[maxn],p[maxn],n,m,T,vis[maxn];
int main()
{
    scanf("%d",&T);
    while(T--)
    {
        memset(vis,0,sizeof(vis));
        scanf("%d%d",&n,&m);
        for(int i=1;i<=n;i++)scanf("%d",&a[i]);
        for(int i=1;i<=m;i++)
        {
            scanf("%d",&p[i]);
            vis[p[i]]=1;
        }
        int sign=0;
        for(int i=1;i<=n;i++)
        {
            if(sign==1)break;
            for(int j=1;j<=n-i;j++)
            {
                if(a[j]>a[j+1])
                {
                    if(!vis[j])
                    {
                        sign=1;
                        break;
                    }
                    else 
                    {
                        swap(a[j],a[j+1]);
                    }
                }
            }
        }
        for(int i=1;i<=n-1;i++)
        {
            if(sign==1)break;
            if(a[i]>a[i+1])
            {
                sign=1;
                break;
            }
        }
        if(sign==1)printf("NO\n");
        else printf("YES\n");
    }
    return 0;
}

C：题目名称：

Perform the Combo

题目大意：

一台机器准备要打印一篇文章，有 m 个打印操作

给出操作序列 p1,p2, …,pm和 长度为 n的字符串 s
对于每个 1≤i≤m，pi表示该机器将位置1∼pi上的所有字母打印了出来
最后，在所有操作做完后，该机器又将整个字符串 s打印了出来

请求出字母表中每个字母在文章中出现了多少次

题解：

先统计位置i之前26个字母每个字母的总数，然后直接模拟求总和，注意最后要加上一次完整的计数。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=200000+5;
int T,n,m,s[maxn][27],ans[27];char a[maxn];
int main()
{
    scanf("%d",&T);
    while(T--)
    {
        memset(ans,0,sizeof(ans));
        scanf("%d%d",&n,&m);
        scanf("%s",a+1);
        for(int i=1;i<=n;i++)
        {
            int x=a[i]-'a'+1;
            for(int j=1;j<=26;j++)
            {
                if(j==x)s[i][j]=s[i-1][j]+1;
                else s[i][j]=s[i-1][j];
            }
        }
        for(int i=1;i<=m;i++)
        {
            int x;scanf("%d",&x);
            for(int j=1;j<=26;j++)
            {
                ans[j]=ans[j]+s[x][j];
            }
        }
        for(int i=1;i<=26;i++)ans[i]=ans[i]+s[n][i];
        for(int i=1;i<=26;i++)printf("%d ",ans[i]);
        cout<<endl;
    }
    return 0;
}

D:题目名称：

Three Integers

题目大意：

题目给出三个整数a≤b≤c。

在一次操作中，你可以给这三个数中任意一个数 +1 或 -1。你可以用同一个数字做这样的操作任意次。但你要保证，变换过后的数，是一个正数。

你想要用最小操作步数使得三个整数 A≤B≤C能够符合：B 能被 A 整除， C能被 B整除。

你需要回答 t 组询问。

题解：

一开始想着找出a与b，a与c的关系，结果这样写有点麻烦。b作为中间数字，当然用它来做媒介最好。注意b可以减到a，也可以加到c，所以枚举的大小应该为max(b-a,c-b);找到b-i的所有因子，得到a改变成b-i的因子的最小变化值ned1，并且记录a变化后的值。再找b-i的所有倍数，得到c变成b-i的倍数的最小变化之ned2，并且记录c变化后的值。再更新总变化值，以及记录a，b，c的值。然后同理求b+i的情况。如果b-i<=0直接去掉。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
const int maxn=10000+5;
int T,A,B,C,a,b,c;
int main()
{
    scanf("%d",&T);
    while(T--)
    {
        int ans=0x3f3f3f3f;
        scanf("%d%d%d",&A,&B,&C);
        for(int i=0;i<=max(B-A,C-B);i++)//注意取max 
        {
            int B1=B-i,ned1=0x3f3f3f3f,ned2=0x3f3f3f3f,nowa,nowc;
            if(B1>0)
            {
                for(int j=1;j<=sqrt(B1);j++)
                {
                    if(B1%j==0)
                    {
                        if(ned1>abs(A-j))
                        {
                            ned1=abs(A-j);
                            nowa=j;
                        }
                        if(ned1>abs(A-B1/j))
                        {
                            ned1=abs(A-B1/j);
                            nowa=B1/j;
                        }
                    }
                }
                for(int j=1;j<=C/B1+1;j++)
                {
                    if(ned2>abs(C-B1*j))
                    {
                        ned2=abs(C-B1*j);
                        nowc=B1*j;
                    }
                }
                if(ans>ned1+ned2+i)
                {
                    ans=ned1+ned2+i;
                    b=B1;a=nowa;c=nowc;
                }    
            }


            B1=B+i,ned1=0x3f3f3f3f,ned2=0x3f3f3f3f;
            for(int j=1;j<=sqrt(B1);j++)
            {
                if(B1%j==0)
                {
                    if(ned1>abs(A-j))
                    {
                        ned1=abs(A-j);
                        nowa=j;
                    }
                    if(ned1>abs(A-B1/j))
                    {
                        ned1=abs(A-B1/j);
                        nowa=B1/j;
                    }
                }
            }
            for(int j=1;j<=C/B1+1;j++)
            {
                if(ned2>abs(C-B1*j))
                {
                    ned2=abs(C-B1*j);
                    nowc=B1*j;
                }
            }
            if(ans>ned1+ned2+i)
            {
                ans=ned1+ned2+i;
                b=B1;a=nowa;c=nowc;
            }
        }
        printf("%d\n",ans);
        printf("%d %d %d\n",a,b,c);
    }
    return 0;
}

E:题目名称：

Construct the Binary Tree

题目大意：

要求构造一个n个节点的二叉树（每个节点拥有不超过2个孩子），节点1为根，要使所有节点到根的距离之和为d。要求先判断可不可以构造，如果可以输出“YES”，下一行输出2到n号节点的父亲节点，否则输出“NO”。有多组询问。

题解：

要想构成树，那么必须保证Minn<=d<=Maxx,Minn就是满二叉树(最接近满二叉树)的情况，Maxx就是链。然后就是从Minn转换或者从Maxx转换成d。这里选择将链转换变成满足深度总和满足d的情况，预处理出Minn[i]，即n=i，对应的最小深度和，若dMaxx那么就NO,(Maxx=i*(i-1)/2)，然后构成链，a[i][0]=i+1,f[i+1]=i,cnt[i]=1;(cnt[i]表示i这个深度点的总数)，从最底层向上移动，移动最多就是和根(1)相接，设最小深度为minn=1，每一次上升的高度为delt=min(i-minn,t)(i为当前移动点的深度，t为需要上升的总次数)。于是点(i+1)移动到深度为i-delt的层，该层最多有2^(i-delt-1)个点，点(i+1）就位于cnt[pos]的位置，f[i+1]对应的是上一层的第cnt[pos]/2的点。即f[i+1]=a[pot-1][cnt[pos]/2]，那么a[pos][cnt[pos]]=i+1;注意每次cnt[pos]++,一旦cnt[pos]>=pow(2,minn)，那么就要往下一层走了。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
const int maxn=5000+5;
int T,n,d,sum[maxn],maxx[maxn],f[maxn],a[maxn][maxn],cnt[maxn];
int main()
{
    scanf("%d",&T);
    int opt=2;
    sum[1]=0;maxx[1]=0;
    for(int i=2;i<=5000;i++)
    {
        maxx[i]=i*(i-1)/2;
        if(i<pow(2,opt))
        {
            sum[i]=sum[i-1]+(opt-1);
        }
        else
        {
            opt++;
            sum[i]=sum[i-1]+(opt-1);
        }
    }
    while(T--)
    {
        scanf("%d%d",&n,&d);
        if(d<sum[n]||d>maxx[n])
        {
            printf("NO\n");
            continue;
        }
        int t=maxx[n]-d;
        for(int i=0;i<=n-1;i++)
        {
            a[i][0]=i+1;f[i+1]=i;cnt[i]=1;
        }
        int minn=1;
        for(int i=n-1;i>=1;i--)
        {
            int delt=min(i-minn,t);
            t-=delt;
            int pos=i-delt;
            f[i+1]=a[pos-1][cnt[pos]/2];
            a[pos][cnt[pos]++]=i+1;
            if(cnt[pos]>=pow(2,minn))minn++;
            if(!t)break;
        }
        printf("YES\n");
        for(int i=2;i<=n;i++)
        {
            printf("%d ",f[i]);
        }
        cout<<endl;
    }
    return 0;
}

F:题目名称：

Moving Points

题目大意：

题解：

假如xivj。不妨让所有的点按xi从小到大的顺序排列。对于i，需要找到所有满足vj<vi的点，这样的i与j的距离就是0，其余的就是abs(xi-xj)，这样就需要维护这种点的vj的总和。可以选择树状数组来维护:先将所有点的vi从d小到大编号，1为最小，n为最大，找到点i的vi的排名p，然后加入到树状数组。排名小于p的就是vj<vp的，否则就是vp<vj的。树状数组要维护点p之前(包括点p)的点的个数和点p之前(包括点p)的v的总和。然后开始枚举点i，首先确定它v的排名p，得到p之前点的个数f1(即v<vi的点的个数)，得到p之前的点的x总和f2(即v<vi的x总和),求得可以使自己和点i之间的距离为0的个数，其余就是不可为0的。ans=ans+abs((n-f1-i)*a[i].x-(sum-f2));注意要减去a[i].x本身，并且每一对点之间只能计算一次，所以树状数组要更新，将权值加上负值即可(add(p,-1,-a[i].x);)。所以剩余距离不能为0的点个数为n-f1-i，而这些点的x之和就是sum-f2(每次sum-a[i].x)。最后输出ans。

开longlong保平安

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const long long maxn=200000+5;
long long n,b[maxn],w1[maxn],w2[maxn];
struct node
{
    long long x,v;
}a[maxn];
long long cmp(node x1,node x2)
{
    return x1.x<x2.x;
}
long long lowbit(long long x)
{
    return x&(-x);
}
void add(long long x,long long v1,long long v2)
{
    while(x<=n)
    {
        w1[x]+=v1;w2[x]+=v2;x+=lowbit(x);
    }
}
long long query1(long long x)
{
    long long ans=0;
    while(x)
    {
        ans+=w1[x];x-=lowbit(x);
    }
    return ans;
}
long long query2(long long x)
{
    long long ans=0;
    while(x)
    {
        ans+=w2[x];x-=lowbit(x);
    }
    return ans;
}
int main()
{
    scanf("%lld",&n);
    long long sum=0;
    for(long long i=1;i<=n;i++)scanf("%lld",&a[i].x);
    for(long long i=1;i<=n;i++)
    {
        scanf("%lld",&a[i].v);
        b[i]=a[i].v;
    }
    sort(a+1,a+1+n,cmp);
    sort(b+1,b+1+n);
    long long tot=unique(b+1,b+1+n)-b-1;
    for(long long i=1;i<=n;i++)
    {
        long long p=lower_bound(b+1,b+1+tot,a[i].v)-b;
        add(p,1,a[i].x);
        sum+=a[i].x;
    } 
    long long ans=0;
    for(long long i=1;i<=n;i++)
    {
        long long p=lower_bound(b+1,b+1+tot,a[i].v)-b;
        sum-=a[i].x;
        long long f1=query1(p-1),f2=query2(p-1);
        ans=ans+abs((n-f1-i)*a[i].x-(sum-f2));
        add(p,-1,-a[i].x);
    }
    printf("%lld\n",ans);
    return 0;
}
