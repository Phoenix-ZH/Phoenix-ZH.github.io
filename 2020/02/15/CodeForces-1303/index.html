A：题目名称：

Erasing Zeroes

题目大意：

删除最少的0，使所有的1连续，问删除多少个0？

题解：

找到最后一个1的位置，和开头1的位置，枚举中间0的个数

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int T;char s[105];
int main()
{
    scanf("%d",&T);
    while(T--)
    {
        scanf("%s",s);
        int sign=0,ans=0,t=0;
        for(int i=strlen(s);i>=0;i--)
        {
            if(s[i]=='1')
            {
                t=i;
                break;
            }
        }
        for(int i=0;i<=t;i++)
        {
            if(!sign)
            {
                if(s[i]=='1')
                {
                    sign=1;
                }
            }
            else if(sign==1) 
            {
                if(s[i]=='0')
                {
                    ans++;
                }
            }
        }
        if(sign==0)ans=0;
        printf("%d\n",ans);
    }
    return 0;
}

B：题目名称：

National Project

题目大意：

修一条长n的路，连续g天修的好路，连续b天修的坏路，好路至少占一半，问要修完这条路至少几天？(一天只修长度1的路)

题解：

先特判：n如果<=g||(n/2<=g&&n>=g)直接修n天

分n为奇数和偶数判断，先找到修g天好路需要t个轮回。如果好路没修够，就需要延长1个轮回，这个轮回不一定完整。如果好路修够了，那么只需要看剩下多长没修，注意期间可以修坏路，最后求个总和。对于延长一个轮回的情况，要加上继续修好路的天数。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
long long T,n,g,b;
int main()
{
    scanf("%lld",&T);
    while(T--)
    {
        scanf("%lld%lld%lld",&n,&g,&b);
        long long p;
        if(n%2==0)p=n/2;
        else p=n/2+1;
        if(n<=g)
        {
            printf("%lld\n",n);
            continue;
        }
        if(p<=g&&n>=g)
        {
            printf("%lld\n",n);
            continue;
        }
        long long ans=0;
        long long t=p/g;
        long long yu=p-t*g;
        if(yu==0)
        {
            long long t2=min(n-p,(t-1)*b);
            long long yu2=n-p-t2;
            ans=yu2+t*g+(t-1)*b;
        }
        else 
        {
            long long t2=min(n-p,t*b);
            long long yu2=n-p-t2;
            ans=yu2+t*(b+g)+yu;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
