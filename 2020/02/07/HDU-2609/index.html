题目名称：

HDU - 2609 How many

题目链接：

https://vjudge.net/problem/HDU-2609

题目大意：

给n个字符串，每个字符串可以滑动形成新的字符串，问有多少个不同的字符串。
ps：若一个字符串滑动后与另一个字符串相同，则这两个字符串相同。

题解：

先将每个字符串的最小表示法找到，然后用string数组储存字符串的最小表示法，然后sort排序，匹配找到总数

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=10000+5;
const int maxm=100+5;
int len2,Next[maxn],len1,n;string s[maxn];char a[maxn][maxm];
int cmp(string x,string y)
{
    return x<y;
}
int get_min(int opt)
{
    int opt1=0,opt2=1,k=0;
    while(opt1<len1&&opt2<len1&&k<len1)
    {
        if(s[opt][(opt1+k)%len1]==s[opt][(opt2+k)%len1])k++;
        else if(s[opt][(opt1+k)%len1]>s[opt][(opt2+k)%len1])
        {
            opt1=opt1+k+1;k=0;
            if(opt1==opt2)
                opt1++;
        }
        else if(s[opt][(opt1+k)%len1]<s[opt][(opt2+k)%len1])
        {
            opt2=opt2+k+1;k=0;
            if(opt1==opt2)
                opt2++;
        }
    }
    return min(opt1,opt2);
}
int main()
{
    while(scanf("%d",&n)!=EOF)
    {
        memset(a,0,sizeof(a));//char  与 string不同，必须清空！string s[i] 清空：s[i].clear(); 
        for(int i=1;i<=n;i++)cin>>s[i];
        for(int i=1;i<=n;i++)
        {
            len1=s[i].length();
            int x=get_min(i);
            int opt=0;
            while(opt<len1)
            {
                a[i][opt]=s[i][(x+opt)%len1];
                opt++;
            }
        }
        for(int i=1;i<=n;i++)s[i]=a[i];
        sort(s+1,s+1+n,cmp);
        int ans=1;
        for(int i=2;i<=n;i++)
        {
            if(s[i]!=s[i-1])ans++;
        }
        printf("%d\n",ans);
    }
    return 0;
}
