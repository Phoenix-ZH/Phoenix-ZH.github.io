题目名称：

HDU - 4292 Food

题目链接：

https://vjudge.net/problem/HDU-4292#author=yupengju

题目大意：

昊妈决定只给每个人一种食物和一种饮料，而且每种饮料和食物都只能提供有限次。现在给出每个人喜欢的食物种类和饮料种类，
昊妈可以最多让多少人得到满足呢？第一行给出N F D表示人数，食物种类数，饮料种类数第二行给出F个数字表示第i种食物有几个第三行给出D个数字
表示第i种饮料有几个接下来N行分别给出第i个人对第j种食物的喜爱情况，Y是喜欢N是不喜欢，再来N行对饮料的

题解：

这道题是最大流问题，重点自然就是建图，一个人要满足，必须同时拥有食物和饮料。而一个人也最多可以满足一次，
而且饮料和食物的容量也是有限的。最简单的自然是食物连向起点，边权为食物的容量；饮料连向中点，边权为饮料的容量；如果某人i喜欢食物j，
那么二者就要连边，即j–>i+N，边权为1，并且人位于中间必须拆点，即i+N–>i+2N,边权为1。如果某人i喜欢饮料j，那么i+2N–>j+3*N，边权为1
建好后跑dinic。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
using namespace std;
const int maxn=200000+5;
const int N=200+5;
int k=1,n,f,d,deep[maxn],head[maxn],ans,s,t;
struct node
{
    int to,next,w;
}edge[maxn*2];
void add(int u,int v,int w)
{
    edge[++k].to=v;edge[k].next=head[u];head[u]=k;edge[k].w=w;
}
bool bfs()
{
    queue<int>q;
    memset(deep,0,sizeof(deep));
    deep[s]=1;
    q.push(s);
    while(!q.empty())
    {
        int x=q.front();q.pop();
        for(int i=head[x];i;i=edge[i].next)
        {
            if(edge[i].w&&!deep[edge[i].to])
            {
                deep[edge[i].to]=deep[x]+1;
                q.push(edge[i].to);
            }
        }
    }
    return deep[t];
}
int dfs(int x,int F)
{
    if(x==t)return F;
    int f=0;
    for(int i=head[x];i;i=edge[i].next)
    {
        if(deep[edge[i].to]==deep[x]+1&&edge[i].w)
        {
            int w=dfs(edge[i].to,min(F-f,edge[i].w));
            edge[i].w-=w;edge[i^1].w+=w;f+=w;
            if(f==F)break;
        }
    }
    if(!f)deep[x]=0;
    return f;
}
void dinic()
{
    while(bfs())
    {
        ans+=dfs(s,0x3f3f3f3f);
    }
}
int main()
{
    while(scanf("%d%d%d",&n,&f,&d)!=EOF)
    {
        k=1;s=0;t=5*N;ans=0;
        memset(head,0,sizeof(head));
        memset(edge,0,sizeof(edge));
        for(int i=1;i<=f;i++)
        {
            int x;scanf("%d",&x);
            add(s,i,x);add(i,s,0);
        }
        for(int i=1;i<=d;i++)
        {
            int x;scanf("%d",&x);
            add(i+3*N,t,x);add(t,i+3*N,0);
        }
        for(int i=1;i<=n;i++)
        {
            string x;
            cin>>x;
            for(int j=1;j<=f;j++)
            {
                if(x[j-1]=='Y')
                {
                    add(j,i+N,1);add(i+N,j,0);
                }
            }
        }
        for(int i=1;i<=n;i++)
        {
            string x;
            cin>>x;
            for(int j=1;j<=d;j++)
            {
                if(x[j-1]=='Y')
                {
                    add(i+2*N,j+3*N,1);add(j+3*N,i+2*N,0);
                }
            }
        }
        for(int i=1;i<=n;i++)
        {
            add(i+N,i+2*N,1);add(i+2*N,i+N,0);
        }
        dinic();
        printf("%d\n",ans);
    }
    return 0;
}
