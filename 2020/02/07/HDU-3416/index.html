题目名称：

HDU-3416 Marriage Match IV

题目链接：

https://vjudge.net/problem/HDU-3416

题目大意：

n 个点 m 条边带权有向图，问每条边最多只能走一次时从 A 到 B 的最短路径数量。

题解：

问题是问从s–>t有多少种完全不同的最短路径。首先要想办法找到哪些边属于最短路径里面的边，于是先正向跑一遍spfa，再反向跑一遍spfa，设smin为s–>t的最短路的长度，那如果一条边属于最短路，那它一定满足dist1[x[i]]+dist2[y[i]]+w[i]==smin&&x[i]!=y[i]。这样就可以找到最短路径的所有边，然后建边时注意，x[i]–>y[i],边权为1，跑dinic，得到的最大流ans就是答案。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
using namespace std;
const int maxn=200010+5;
int T,n,m,s,t,head[maxn],dist2[maxn],dist1[maxn],vis[maxn],k=1,k2=1,head2[maxn],deep[maxn],x[maxn],y[maxn],w[maxn];
struct node
{
    int to,next,w;
}edge[maxn*2];
void add(int u,int v,int w)
{
    edge[++k].to=v;edge[k].next=head[u];head[u]=k;edge[k].w=w;
}
void spfa1(int s)//dist1[i]表示从A--i的距离,spfa1跑正边 
{
    queue<int>q;
    memset(vis,0,sizeof(vis));
    memset(dist1,0x3f3f3f3f,sizeof(dist1));
    dist1[s]=0;vis[s]=1;
    q.push(s);
    while(!q.empty())
    {
        int x=q.front();q.pop();vis[x]=0;
        for(int i=head[x];i;i=edge[i].next)
        {
            if(dist1[edge[i].to]>dist1[x]+edge[i].w)
            {
                dist1[edge[i].to]=dist1[x]+edge[i].w;
                if(!vis[edge[i].to])
                {
                    vis[edge[i].to]=1;
                    q.push(edge[i].to);
                }
            }
        }
    }
}
void spfa2(int s)//dist2[i]表示从i--B的距离,spfa2跑反边 
{
    queue<int>q;
    memset(vis,0,sizeof(vis));
    memset(dist2,0x3f3f3f3f,sizeof(dist2));
    dist2[s]=0;vis[s]=1;
    q.push(s);
    while(!q.empty())
    {
        int x=q.front();q.pop();vis[x]=0;
        for(int i=head[x];i;i=edge[i].next)
        {
            if(dist2[edge[i].to]>dist2[x]+edge[i].w)
            {
                dist2[edge[i].to]=dist2[x]+edge[i].w;
                if(!vis[edge[i].to])
                {
                    vis[edge[i].to]=1;
                    q.push(edge[i].to);
                }
            }
        }
    }
}
bool bfs()
{
    queue<int>q;
    memset(deep,0,sizeof(deep));
    q.push(s);deep[s]=1;
    while(!q.empty())
    {
        int x=q.front();q.pop();
        for(int i=head[x];i;i=edge[i].next)
        {
            if(!deep[edge[i].to]&&edge[i].w)
            {
                deep[edge[i].to]=deep[x]+1;
                q.push(edge[i].to);
            }
        }
    }
    return deep[t];
}
int dfs(int x,int F)
{
    if(x==t)return F;
    int f=0;
    for(int i=head[x];i;i=edge[i].next)
    {
        if(deep[edge[i].to]==deep[x]+1&&edge[i].w)
        {
            int w=dfs(edge[i].to,min(edge[i].w,F));
            edge[i].w-=w;edge[i^1].w+=w;f+=w;F-=w;
        }
    }
    if(!f)deep[x]=0;
    return f;
}
int dinic()
{
    int ans=0;
    while(bfs())
    {
        ans+=dfs(s,0x3f3f3f3f);
    }
    return ans;
}
int main()
{
    scanf("%d",&T);
    while(T--)
    {
        k=1;
        memset(head,0,sizeof(head));
        memset(edge,0,sizeof(edge));
        memset(x,0,sizeof(x));
        memset(y,0,sizeof(y));
        memset(w,0,sizeof(w));
        scanf("%d%d",&n,&m);
        for(int i=1;i<=m;i++)
        {
            int a,b,c;
            scanf("%d%d%d",&x[i],&y[i],&w[i]);
        }
        scanf("%d%d",&s,&t);
        for(int i=1;i<=m;i++)
        {
            add(x[i],y[i],w[i]);
        }
        spfa1(s);
        int smin=dist1[t];
        k=1;
        memset(head,0,sizeof(head));
        memset(edge,0,sizeof(edge));
        for(int i=1;i<=m;i++)
        {
            add(y[i],x[i],w[i]);
        }
        spfa2(t);
        k=1;
        memset(head,0,sizeof(head));
        memset(edge,0,sizeof(edge));
        for(int i=1;i<=m;i++)
        {
        //    cout<<dist1[x[i]]<<' '<<dist2[y[i]]<<' '<<w[i]<<' '<<smin<<endl;
            if(dist1[x[i]]+dist2[y[i]]+w[i]==smin&&x[i]!=y[i])
            {
                add(x[i],y[i],1);add(y[i],x[i],0);
            }
        }
        int ans=dinic();
        printf("%d\n",ans);
    }
    return 0;
}
