题目名称：

POJ - 3268 Silver Cow Party

题目链接：

https://vjudge.net/problem/POJ-3268#author=Demon_de_Laplace

题目大意：

N头牛，M条单行道，在s出举行聚会，每头牛出发和返回都是走最短路，问所有牛中走的路程最远的牛走了多远？

题解：

建边的时候建双向边，k=2开始.
当边号为偶数时，说明是正边；
当边号为奇数时，说明时反边；
从s出发，跑正边，dist[i]存的是从s到i的最短距离；
从s出发，跑反边，dist[i]存的是从i到s的最短距离；
ans=max(ans,dist1[i]+dist2[i])

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
using namespace std;
const int maxn=100000+5;
int n,m,t,k=1,head[maxn],vis[maxn],dist1[maxn],dist2[maxn];
struct node
{
    int to,next,w;
}edge[maxn*2];
void add(int u,int v,int w)
{
    edge[++k].to=v;edge[k].next=head[u];head[u]=k;edge[k].w=w;
}
void spfa1()
{
    queue<int>q;
    memset(vis,0,sizeof(vis));
    memset(dist1,0x3f3f3f3f,sizeof(dist1));
    vis[t]=1;dist1[t]=0;
    q.push(t);
    while(!q.empty())
    {
        int x=q.front();q.pop();vis[x]=0;
        for(int i=head[x];i;i=edge[i].next)
        {
            if(i%2==1)continue;
            if(dist1[edge[i].to]>dist1[x]+edge[i].w)
            {
                dist1[edge[i].to]=dist1[x]+edge[i].w;
                if(!vis[edge[i].to])
                {
                    vis[edge[i].to]=1;
                    q.push(edge[i].to);
                }
            }
        }
    }
}
void spfa2()
{
    queue<int>q;
    memset(vis,0,sizeof(vis));
    memset(dist2,0x3f3f3f3f,sizeof(dist2));
    vis[t]=1;dist2[t]=0;
    q.push(t);
    while(!q.empty())
    {
        int x=q.front();q.pop();vis[x]=0;
        for(int i=head[x];i;i=edge[i].next)
        {
            if(i%2==0)continue;
            if(dist2[edge[i].to]>dist2[x]+edge[i].w)
            {
                dist2[edge[i].to]=dist2[x]+edge[i].w;
                if(!vis[edge[i].to])
                {
                    vis[edge[i].to]=1;
                    q.push(edge[i].to);
                }
            }
        }
    }
}
int main()
{
    scanf("%d%d%d",&n,&m,&t);
    for(int i=1;i<=m;i++)
    {
        int a,b,c;scanf("%d%d%d",&a,&b,&c);
        add(a,b,c);add(b,a,c);
    }
    spfa1();spfa2();
    int ans=0;
    for(int i=1;i<=n;i++)
    {
        ans=max(ans,dist1[i]+dist2[i]);
    }
    printf("%d\n",ans);
    return 0;
}
