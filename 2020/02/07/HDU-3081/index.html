题目名称：

HDU - 3081 Marriage Match II

题目链接：

https://vjudge.net/problem/HDU-3081

题目大意：

有N个女孩要与N个男孩玩配对游戏.每个女孩有一个可选男孩的集合(即该女孩可以选自己集合中的任意一个男孩作为该轮的搭档).
然后从第一轮开始,每个女孩都要和一个不同的男孩配对.如果第一轮N个女孩都配对成功,那么就开始第二轮配对,女孩依然从自己的备选男孩集合中选择,
但是不能选那些已经被该女孩在前几轮选择中选过的男孩了(比如i女孩在第一轮选了j男孩,那么i在第二轮就不能选j男孩了). 问你游戏最多能进行多少轮?

题解：

女生:1~n;男生:(n+1)~(n*2)。对于女生u，她可以直接连向不讨厌的男生，边权为1。

假设我们当前二分尝试的轮数为K(即能够进行K轮匹配):首先如果女孩i可能选择男孩j,那么就有边(i, j+n, 1).且源点到每个女孩i有边(s,i,K),
每个男孩j到汇点t有边(j+n,t,K).

如果最大流==K*n,那么就表示可以进行最少K轮匹配.

证明：如果满流,那么每个女生肯定选择了K个不同的男孩,每个男孩肯定被K个不同的女孩选择了(因为一个女孩到一个男孩边容量只为1,所以该女孩最多只能选该男孩一次).
那么上面这样就能保证这个游戏可以进行K轮吗?可以的,假设当前图的流量为0,说明任何女孩都没选男孩. 你可以想象假如此时从S到所有女孩有流量1
(虽然容量是K,但是目前我们只放出1流量)流出,那么这些流量肯定会汇集到t(因为最大流为Kn,而我们此时只不过n流量).
这个汇集的过程就是第一轮女孩选择了各自不同男孩的结果. 现在从S到所有女孩又有流量1流出(即第二轮开始了),
这些流量肯定又经过了n个男孩汇集到t点了 且 如果上一轮i女孩的流量走到j男孩,这一轮i女孩的流量肯定不走j男孩了(因为i女孩到j男孩的边只有1容量).
综上所述,只要最大流==Kn,那么就能进行K轮.

即如果女孩{1,2,3,4,5,6,7,8,9}是朋友的话,且2号女孩能选择男孩3,我们如果标记来使得其他所有女孩都能选择男孩3呢?
我选择的是闭包传递，当然也可以用并查集。
ps：不要把男孩也算进,比如女孩1与男孩2能选,女孩3与男孩2能选,但是女孩1与女孩3可不是朋友,她们不属于同一个并查集。
一开始这里就出了问题。如果我们dist[a][b+N]=1,而dist[b+N][a]不管，就不存在问题了！

代码

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
using namespace std;
const int maxn=200000+5;
const int N=200+5;
int k=1,head[maxn],deep[maxn],s,t,n,m,ans,T,f,dist[N][N];
struct node
{
    int to,next,w;
}edge[maxn*2];
void add(int u,int v,int w)
{
    edge[++k].to=v;edge[k].next=head[u];head[u]=k;edge[k].w=w;
}
bool bfs()
{
    queue<int>q;
    memset(deep,0,sizeof(deep));
    deep[s]=1;
    q.push(s);
    while(!q.empty())
    {
        int x=q.front();q.pop();
        for(int i=head[x];i;i=edge[i].next)
        {
            if(!deep[edge[i].to]&&edge[i].w)
            {
                deep[edge[i].to]=deep[x]+1;
                q.push(edge[i].to);
            }
        }
    }
    return deep[t];
}
int dfs(int x,int F)
{
    if(x==t)return F;
    int f=0;
    for(int i=head[x];i;i=edge[i].next)
    {
        if(deep[edge[i].to]==deep[x]+1&&edge[i].w)
        {
            int w=dfs(edge[i].to,min(F-f,edge[i].w));
            edge[i].w-=w;edge[i^1].w+=w;f+=w;
            if(f==F)break;
        }
    }
    if(!f)deep[x]=0;
    return f;
}
void dinic()
{
    while(bfs())
    {
        ans+=dfs(s,0x3f3f3f3f);
    }
}
bool solve(int mid)
{
        k=1;
        memset(head,0,sizeof(head));
        memset(edge,0,sizeof(edge));
        ans=0;
        for(int i=1;i<=n;i++)
        {
            add(s,i,mid);add(i,s,0);
        }
        for(int i=1;i<=n;i++)
        {
            for(int j=n+1;j<=2*n;j++)
            {
                if(dist[i][j]==0x3f3f3f3f)continue;
                add(i,j,1);add(j,i,0);
            }
        }
        for(int i=1+n;i<=n*2;i++)
        {
            add(i,t,mid);add(t,i,0);
        }
        dinic();
        if(ans==mid*n)return 1;
        return 0;
}
int main()
{
    //1-ngirl,(n+1)-(2*n)boy
    scanf("%d",&T);
    while(T--)
    {
        k=1;
        memset(head,0,sizeof(head));
        memset(edge,0,sizeof(edge));
        ans=0;
        s=0,t=8*N;
        scanf("%d%d%d",&n,&m,&f);
        memset(dist,0x3f3f3f3f,sizeof(dist));
        for(int i=1;i<=m;i++)
        {
            int a,b;scanf("%d%d",&a,&b);
            dist[a][b+n]=1;
        }
        for(int i=1;i<=f;i++)
        {
            int a,b;scanf("%d%d",&a,&b);
            dist[a][b]=dist[b][a]=1;
        }
        for(int K=1;K<=2*n;K++)
        {
            for(int i=1;i<=2*n;i++)
            {
                for(int j=1;j<=2*n;j++)
                {
                    if(dist[i][K]==1&&dist[K][j]==1)
                    {
                        dist[i][j]=1;
                    }
                }
            }
        }
        int l=0,r=n,ANS=0;
        while(l<=r)
        {
            int mid=(l+r)/2;
            if(solve(mid))
            {
                ANS=mid;
                l=mid+1;
            }
            else r=mid-1;
        }
        printf("%d\n",ANS);
    }
    return 0;
}
