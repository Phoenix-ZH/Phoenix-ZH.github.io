题目名称：

POJ - 1087 A Plug for UNIX

题目链接：

https://vjudge.net/problem/POJ-1087#author=cjyl

题目大意：

宾馆里有几种插座，分别有其对应型号，你携带了几种用电器（手机，电脑一类的），也有其对应型号；
可是不一定用电器就能和插座匹配上，于是宾馆的商店里提供了一些转换器，这些转换器可以将某一型号电源转换成另一型号的。
问，你的用电器最少会有多少种无法充电 。注意插座型号是字符串形式，长度小于24，用电器名称也是字符串。

题解：

这是最大流问题。先说一下细节，插座和用电器的数量肯定都只有一个，但是转换器的数量是无限的。
建图：设起点为s，终点为t，首先s与所有的用电器相连，边权为1，（一开始还想着拆点，我怕是个哈批啊，对于两边的点，边权就是容量，
不需要去拆点！），t与所有的插座相连。如果型号相同，就可以二者之间进行连边。然后对于转换器，转换器之间可以相连，并且转换可以实现A–>B，
也可以实现B–>A。所以，对于用电器型号=A||B的都可以连边，对于插座型号=A||B的都可以连边，边权为1，对于转换器之间可以相连的，直接连接，
由于容量无穷大，因此不必拆点！

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
#include<map>
using namespace std;
const int maxn=300000+5;
const int N=200+5;
int s,t,n,k=1,m,q,ans,head[maxn],deep[maxn];char r[maxn][25],mr[maxn][25],mc[maxn][25];
struct node
{
    int to,next,w;
}edge[maxn*2];
struct N
{
    char s[40];char r[25];
}a[maxn];
void add(int u,int v,int w)
{
    edge[++k].to=v;edge[k].next=head[u];head[u]=k;edge[k].w=w;
}
bool bfs()
{
    queue<int>q;
    memset(deep,0,sizeof(deep));
    deep[s]=1;
    q.push(s);
    while(!q.empty())
    {
        int x=q.front();q.pop();
        for(int i=head[x];i;i=edge[i].next)
        {
            if(!deep[edge[i].to]&&edge[i].w)
            {
                deep[edge[i].to]=deep[x]+1;
                q.push(edge[i].to);
            }
        }
    }
    return deep[t];
}
int dfs(int x,int F)
{
    if(x==t)return F;
    int f=0;
    for(int i=head[x];i;i=edge[i].next)
    {
        if(deep[edge[i].to]==deep[x]+1&&edge[i].w)
        {
            int w=dfs(edge[i].to,min(F,edge[i].w));
            edge[i].w-=w;edge[i^1].w+=w;f+=w;F-=w;
        }
    }
    if(!f)deep[x]=0;
    return f;
}
void dinic()
{
    ans=0;
    while(bfs())
    {
        ans+=dfs(s,0x3f3f3f3f);
    }
}
int main()
{
    while(scanf("%d",&n)!=EOF)
    {
        k=1;
        memset(head,0,sizeof(head));
        memset(edge,0,sizeof(edge));
        s=0;t=4*N;
        for(int i=1;i<=n;i++)
        {
            scanf("%s",r[i]);
        }
        scanf("%d",&q);
        for(int i=1;i<=q;i++)
        {
            add(s,i,1);add(i,s,0);
            char x;
            scanf("%s%s",a[i].s,a[i].r);
        }
        for(int i=1;i<=q;i++)
        {
            for(int j=1;j<=n;j++)
            {
                if(strcmp(a[i].r,r[j])==0)
                {
                    add(i,j+N,1);add(j+N,i,0);
                }
            }
        }
        for(int i=1;i<=n;i++)
        {
            add(i+N,t,1);add(t,i+N,0);
        }
        scanf("%d",&m);
        for(int i=1;i<=m;i++)
        {
            scanf("%s%s",mr[i],mc[i]);
            for(int j=1;j<=q;j++)
            {
                if(strcmp(a[j].r,mr[i])==0||strcmp(a[j].r,mc[i])==0)
                {
                    add(j,i+2*N,1);add(i+2*N,j,0);
                }
            }
            for(int j=1;j<=n;j++)
            {
                if(strcmp(r[j],mc[i])==0||strcmp(r[j],mr[i])==0)
                {
                    add(i+2*N,j+N,1);add(j+N,i+2*N,0);
                }
            }
        }
        for(int i=1;i<=m;i++)
        {
            for(int j=1;j<=m;j++)
            {
                if(i!=j&&strcmp(mc[i],mr[j])==0)
                {
                    add(i+2*N,j+2*N,0x3f3f3f3f);add(j+2*N,i+2*N,0);
                }
            }
            for(int j=1;j<=m;j++)
            {
                if(i!=j&&strcmp(mr[i],mc[j])==0)
                {
                    add(i+2*N,j+2*N,0x3f3f3f3f);add(j+2*N,i+2*N,0);
                }
            }
        }
        dinic();
        printf("%d\n",q-ans);
    }
    return 0;
}
