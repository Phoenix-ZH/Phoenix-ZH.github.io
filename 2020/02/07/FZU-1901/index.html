题目名称：

FZU - 1901 Period II

题目链接：

https://vjudge.net/problem/FZU-1901

题目大意：

对于一个正整数p，p<=字符串的长度，使得任意的s[i]==s[i+p]，求出所有符合要求的p。

题解：

这个题和之前的某个题意很迷的密码题有点相似，那个题当时就是用的ex-kmp，做这个题开始时题意不理解，百度了一下题意。一开始想的就是用kmp，似乎网上用kmp的很多，但我一下子不知道如何运行Next，不过，我突然发现，令s2=s1，用s1的所有后缀匹配s2的结果让我意外，所有符合Extend[i]+i>=len的都是p，当然len本身也是。很显然这个题可以用ex-kmp来做。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=1000000+5;
int T,len1,len2,Next[maxn],Extend[maxn],Ans[maxn];string s1,s2;
//s1的所有后缀与s2匹配 
void next_pre()
{
    int i=0,po=0;
    while(s2[i]==s2[i+1]&&i+1<len2)i++;
    Next[0]=len2;Next[1]=i;po=1;
    for(int i=2;i<len2;i++)
    {
        if(Next[i-po]+i<Next[po]+po)
            Next[i]=Next[i-po];
        else 
        {
            int j=Next[po]+po-i;
            if(j<0)j=0;
            while(s2[j+i]==s2[j]&&i+j<len2)j++;
            Next[i]=j;
            po=i;
        }
    }
} 
void extend_pre()
{
    int i=0,po=0;
    while(s1[i]==s2[i]&&i<len2&&i<len1)i++;
    Extend[0]=i;
    for(int i=1;i<len1;i++)
    {
        if(Next[i-po]+i<Extend[po]+po)
            Extend[i]=Next[i-po];
        else 
        {
            int j=Extend[po]+po-i;
            if(j<0)j=0;
            while(s1[j+i]==s2[j]&&i+j<len1&&j<len2)j++;
            Extend[i]=j;
            po=i;
        }
    }
}

int main()
{
    scanf("%d",&T);int now=0;
    while(now<T)
    {
        now++;
        memset(Next,0,sizeof(Next));
        memset(Extend,0,sizeof(Extend));
        memset(Ans,0,sizeof(Ans));
        cin>>s2;s1=s2;
        len2=s2.length();len1=s1.length();
        next_pre();
        extend_pre();
        int opt=0,res=0;int ans=0;
        for(int i=0;i<len2;i++)
        {
            if(Extend[i]+i>=len2&&i!=0)
            {
                Ans[++ans]=i;
            }    
        }
        Ans[++ans]=len2;
        printf("Case #%d: %d\n",now,ans);
        for(int i=1;i<ans;i++)printf("%d ",Ans[i]);printf("%d\n",Ans[ans]);
    }
    return 0;
}
//注意数组不能太大，容易TLE 
//一开始找到第一个Extend[i]+i>=len2就以为找到循环节了，实际上有问题。要线性枚举一遍才行！
