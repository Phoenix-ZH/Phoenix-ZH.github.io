题目名称：

HDU - 3294 Girls’ research

题目链接：

https://vjudge.net/problem/HDU-3294#author=asasas5d

题目大意：

先给一个字符x，然后给一个 字符串，表示字符串中所有字符x变成’a’，其余的字符依次往前变换。例如b babad就转换成azazc。让寻找第一个最长回文串，长度若小于2，则输出No solution!，否则输出起点，终点和转换后的最长回文串。

题解：

最长回文串问题，当然就是用Manacher，先计算出字符x与‘a’的差值，然后将字符串进行转换。然后直接跑Manacher，更新ans时注意更新储存一下终点和起点

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=200000*2+105;
int p[maxn],len,a[maxn],S,T;char s[maxn],t[maxn],x[3];
int main()
{
    while(scanf("%s",x)!=EOF)
    {
        len=0;S=0;T=0;
        scanf("%s",&s);
        int delt=x[0]-'a';
        int L=strlen(s);
        for(int i=0;i<L;i++)
        {
            a[i]=s[i]-'a'-delt;
            if(a[i]<0)a[i]+=26;
            s[i]=a[i]+'a';
        }
        t[0]='$';
        for(int i=1;i<=L;i++)
        {
            t[++len]='#';t[++len]=s[i-1];
        }
        t[++len]='#';
        int mx=0,id=0,ans=0;
        for(int i=1;i<=len;i++)
        {
            if(i<mx)p[i]=min(p[id*2-i],mx-i);
            else p[i]=1;
            while(t[i+p[i]]==t[i-p[i]])p[i]++;
            if(i+p[i]>mx)
            {
                mx=i+p[i];
                id=i;
                if(ans<p[i]-1)
                {
                    ans=p[i]-1;
                    T=(mx-1)/2;
                    S=T-ans+1;
                    T--;S--;
                }
            }
        }
        if(ans<2)
        {
            printf("No solution!\n");
        }
        else 
        {
            printf("%d %d\n",S,T);
            for(int i=S;i<=T;i++)
            printf("%c",s[i]);cout<<endl;
        }
    }
    return 0;
}
