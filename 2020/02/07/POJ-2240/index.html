题目名称：

POJ - 2240 Arbitrage

题目链接：

https://vjudge.net/problem/POJ-2240#author=Andy_acmer

题目大意：

有n种货币，m条货币的汇率关系。比如：USDollar 0.5 BritishPound
说明1个USDollar货币，可以换成0.5个BritishPound货币，问是否可以通过这种关系来获利。

题解：

由于货币名字使字符串，那么用map比较方便,map<string,int>q，可以将字符串映射成int型。然后跑spfa，看是否存在正环，与判断负环的方法一样。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
#include<map>
using namespace std;
const int maxn=10000+5;
int n,m,head[maxn],vis[maxn],used[maxn],k=1;double dist[maxn];
map<string,int>q;
struct node
{
    int to,next;double w;
}edge[maxn*2];
void add(int u,int v,double w)
{
    edge[++k].to=v;edge[k].next=head[u];head[u]=k;edge[k].w=w;    
}
int spfa()
{
    queue<int>q;
    memset(dist,0,sizeof(dist));
    memset(vis,0,sizeof(vis));
    memset(used,0,sizeof(used));
    vis[1]=1;used[1]=1;dist[1]=1.0;
    q.push(1);
    while(!q.empty())
    {
        int x=q.front();q.pop();vis[x]=0;
        for(int i=head[x];i;i=edge[i].next)
        {
            if(dist[edge[i].to]<dist[x]*edge[i].w)
            {
                dist[edge[i].to]=dist[x]*edge[i].w;
                if(!vis[edge[i].to])
                {
                    vis[edge[i].to]=1;
                    used[edge[i].to]++;
                    if(used[edge[i].to]>n)return 1;
                    q.push(edge[i].to);
                }
            }
        }
    }
    return 0;
}
int main()
{
    int now=0;
    while(1)
    {
        now++;
        k=1;memset(head,0,sizeof(head));
        memset(edge,0,sizeof(edge));
        q.clear();
        scanf("%d",&n);
        if(n==0)break;
        int tot=0;
        for(int i=1;i<=n;i++)
        {
            string x;cin>>x;
            q[x]=++tot;    
        }        
        scanf("%d",&m);
        for(int i=1;i<=m;i++)
        {
            string a,b;double c;
            cin>>a>>c>>b;
            add(q[a],q[b],c);
        }
        int x=spfa();
        printf("Case %d: ",now);
        if(x==1)printf("Yes\n");
        else printf("No\n");
    }
    return 0;
}

最短路 判环
