题目名称：

HDU - 2732 Leapin’ Lizards

题目链接：

https://vjudge.net/problem/HDU-2732

题目大意：

题目是说一个n*m的迷宫中，有每个格子有柱子。柱子高度为0~3，高度为0的柱子是不能站的（高度为0就是没有柱子）
在一些有柱子的格子上有一些蜥蜴，一次最多跳距离d,相邻格子的距离是1，只要跳出迷宫就是安全的。这个距离是曼哈顿距离（好像是的）
。蜥蜴一次最多跳距离d,但是起跳的地方的柱子高度会减一，一个柱子同一时间只能有一个蜥蜴要求最少几个不能逃出迷宫。

题解：

最大流问题，求出dinic跑出的最大流，然后总数-ans就是剩余的数。建图方法：

1.每个柱子有容纳量，超过了就塌了。u–>u’ 边权为容纳量

2.有蜥蜴的柱子与起点s相连，即s–>u,边权为1

3.边界上可以跳出界的与终点相连，条件(x1<=p||y1<=p||n-x1+1<=p||len-y1+1<=p)&&h[x1][y1]，u’–>t,边权0x3f3f3f3f

4.任意两个有容纳量的柱子，且曼哈顿距离<=d，可以相连，即u’–>v,v’->u，边权为0x3f3f3f3f.

ps：1.注意输出时候单复数！
2.注意是j+(i-1)*len(len 表示一行的列数！！！）一开始写成了n，WA了一片。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
using namespace std;
const int maxn=100000+5;
const int N=500+5;
int k=1,head[maxn],deep[maxn],s,t,n,m,T,ans,p,h[30][N*10];
struct node
{
    int to,next,w;
}edge[maxn*2];
void add(int u,int v,int w)
{
    edge[++k].to=v;edge[k].next=head[u];head[u]=k;edge[k].w=w;
}
bool bfs()
{
    queue<int>q;
    memset(deep,0,sizeof(deep));
    deep[s]=1;
    q.push(s);
    while(!q.empty())
    {
        int x=q.front();q.pop();
        for(int i=head[x];i;i=edge[i].next)
        {
            if(!deep[edge[i].to]&&edge[i].w)
            {
                deep[edge[i].to]=deep[x]+1;
                q.push(edge[i].to);
            }
        }
    }
    return deep[t];
}
int dfs(int x,int F)
{
    if(x==t)return F;
    int f=0;
    for(int i=head[x];i;i=edge[i].next)
    {
        if(deep[edge[i].to]==deep[x]+1&&edge[i].w)
        {
            int w=dfs(edge[i].to,min(F-f,edge[i].w));
            edge[i].w-=w;edge[i^1].w+=w;f+=w;
            if(f==F)break;
        }
    }
    if(!f)deep[x]=0;
    return f;
}
void dinic()
{
    while(bfs())
    {
        ans+=dfs(s,0x3f3f3f3f);
    }
}
int main()
{
    scanf("%d",&T);
    int opt=0;
    while(T--)
    {
        opt++;
        k=1;ans=0;s=0;t=5*N;
        memset(head,0,sizeof(head));
        memset(edge,0,sizeof(edge));
        memset(h,0,sizeof(h));
        scanf("%d%d",&n,&p);
        int len=0;int tot=0;
        for(int i=1;i<=n;i++)
        {
            string x;
            cin>>x;len=x.length();
            for(int j=1;j<=len;j++)
            {
                int w=x[j-1]-'0';h[i][j]=w;
                if(w)
                {
                    add(j+(i-1)*len,j+(i-1)*len+N,w);add(j+(i-1)*len+N,j+(i-1)*len,0);
                }
            }
        }
        for(int i=1;i<=n;i++)
        {
            string x;
            cin>>x;len=x.length();
            for(int j=1;j<=len;j++)
            {
                if(x[j-1]=='L')
                {
                    tot++;
                    add(s,j+(i-1)*len,1);add(j+(i-1)*len,s,0);
                }
            }
        }
        for(int x1=1;x1<=n;x1++)
        {
            for(int y1=1;y1<=len;y1++)
            {
                for(int x2=1;x2<=n;x2++)
                {
                    for(int y2=1;y2<=len;y2++)
                    {
                        if(x1==x2&&y1==y2)continue;
                        if(abs(x1-x2)+abs(y1-y2)>p)continue;
                        if(!h[x1][y1]||!h[x2][y2])continue;
                        add(y1+(x1-1)*len+N,y2+(x2-1)*len,0x3f3f3f3f);add(y2+(x2-1)*len,y1+(x1-1)*len+N,0);
                        add(y2+(x2-1)*len+N,y1+(x1-1)*len,0x3f3f3f3f);add(y1+(x1-1)*len,y2+(x2-1)*len+N,0);
                    }
                }
            }
        }
        for(int x1=1;x1<=n;x1++)
        {
            for(int y1=1;y1<=len;y1++)
            {
                if((x1<=p||y1<=p||n-x1+1<=p||len-y1+1<=p)&&h[x1][y1])
                {
                    add(y1+(x1-1)*len+N,t,0x3f3f3f3f);add(t,y1+(x1-1)*len+N,0);
                }
            }
        }
        dinic();
        printf("Case #%d: ",opt);
        if(ans==tot)
        {
            printf("no lizard was left behind.\n");
            continue;
        }
        else if(tot-ans==1)
        {
            printf("1 lizard was left behind.\n");
        }
        else 
        {
            printf("%d lizards were left behind.\n",tot-ans);
        }
    }
    return 0;
}
