题目名称：

POJ - 1062 昂贵的聘礼

题目链接：

https://vjudge.net/problem/POJ-1062#author=0

题目大意：

一个人要买1号物品，而此物品可以通过直接购买或者到别人的手中买别的东西+一些金币来得到。但是需要注意：如果两人地位等级差距超过了delt，就不可以进行间接交易，即该交易圈的人两两之间等级差异不可大于delt。问要得到1号物品至少要花费多少钱？

题解：

建立一个源点，它与所有物品建边，即s–>i（边权为物品原价），因为每个物品除了直接购买，还可以间接购买，设间接购买物品为j，则建边方式为j–>i（边权为额外金币）然后跑最短路，注意最短路的条件不再是dist[edge[i].to]>dist[x]+edge[i].w，因为当价格相等时也需要进入队列来进行松弛，所以应变为dist[edge[i].to]>dist[x]+edge[i].w;然后就是等级限制的判断了。刚开始的s的等级需要从最低到最高都设置一次，意味着跑n次spfa，s就是这次spfa的最低等级，因此a[edge[i].to].l>=limit&&edge[i].to]-limit<=delt||x==0||delt==0；delt==0意味着没有等级限制。

代码：

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
using namespace std;
const int maxn=100000+5;
int n,delt,head[maxn],vis[maxn],dist[maxn],k=1,liv[maxn],ans=0x3f3f3f3f;
struct Node
{
    int p,l,x;
}a[maxn];
struct node
{
    int next,to,w;
}edge[maxn*2];
void add(int u,int v,int w)
{
    edge[++k].to=v;edge[k].next=head[u];head[u]=k;edge[k].w=w;
}
void spfa(int limit)
{
    queue<int>q;
    for(int i=1;i<=n;i++)dist[i]=a[i].p;
    memset(vis,0,sizeof(vis));
    vis[0]=1;dist[0]=0;
    q.push(0);
    while(!q.empty())
    {
        int x=q.front();q.pop();vis[x]=0;
        for(int i=head[x];i;i=edge[i].next)
        {
            if(dist[edge[i].to]>=dist[x]+edge[i].w&&(delt==0||(a[edge[i].to].l>=limit&&a[edge[i].to].l-limit<=delt)))
            {
                dist[edge[i].to]=dist[x]+edge[i].w;
                if(!vis[edge[i].to])
                {
                    vis[edge[i].to]=1;
                    q.push(edge[i].to);
                }
            }
        }    
    }
    ans=min(ans,dist[1]);
}
int main()
{
    scanf("%d%d",&delt,&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d%d%d",&a[i].p,&a[i].l,&a[i].x);
        liv[i]=a[i].l;
        add(0,i,a[i].p);
        for(int j=1;j<=a[i].x;j++)
        {
            int x,y;
            scanf("%d%d",&x,&y);
            add(x,i,y);
        }
    }
    sort(liv+1,liv+1+n);
    for(int i=1;i<=n;i++)
    spfa(liv[i]);
    printf("%d\n",ans);
    return 0;
}
