A. Arena

题目大意

比赛次数无限次，赢了的人可以等级+1，问有多少个人可以赢。

思路

枚举a[i]>a1的个数即可

代码

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn=100+50;
int T,n,m,a[maxn];
int main(){
    cin>>T;
    while(T--){
        cin>>n;
        for(int i=1;i<=n;i++)cin>>a[i];
        sort(a+1,a+1+n);
        int tot=0;
        for(int i=2;i<=n;i++){
            if(a[i]>a[1]){
                tot++;
            }
        }
        cout<<tot<<endl;
    }
    return 0;
}

B. Cat Cycle

题目大意

A的顺序是n,n-1,n-2,…,3,2,1,n,n-1…;B的顺序是1,2,3,…,n,1,2…如果某一时刻B的值与A相等，那么B就顺次再移动一次（即B的一小时相当于跳了两格，而正常情况下跳一格）,问在k小时，B的位置是多少？

思路

假如n=5：

时刻:1,2,3,4,5,6,7,8,9…
A：5,4,3,2,1,5,4,3,2,1,5,4,3,2,1,5,4,3,2,1…
B：1,2,4,5,2,3,5,1,3,4,1,2,4,5,2,3,5,1,3,4…

B发生跳跃的时刻是3，5，7，9…

假如n=7：

A：7,6,5,4,3,2,1,7,6,5,4,3,2,1,7,6,5,4,3,2,1…
B：1,2,3,5,6,7,2,3,4,6,7,1,3,4,5,7,1,2,4,5,6…

B发生跳跃的时刻是4 ，7，10，13…

假如n=4：

A:4,3,2,1,4,3,2,1…
B:1,2,3,4,1,2,3,4…

A与B的值不会发生重叠

所以规律就是：如果n是偶数，B正常跳即可；如果n是奇数，B会在(n+1)/2,(n+1)/2+(n+1)/2-1,(n+1)/2*3-2…多跳一格。

代码

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

ll T,n,k;
int main(){
    cin>>T;
    while(T--){
        cin>>n>>k;
        if(n%2==0){
            k%=n;
            if(k==0)k=n;
            cout<<k<<endl;
        }
        else{
            ll t=(n+1)/2-1;
            ll cnt=(k-1)/t;
            k+=cnt;
            k%=n;
            if(k==0)k=n;
            cout<<k<<endl;
        }
    }
    return 0;
}

C. Minimum Ties

题目大意

一共有n(n-1)/2场比赛，输出n(n-1)/2个整数，分别表示1队VS2队，1队VS3队，1队VSn队，2队VS3队的比赛结果，1表示左边的队伍胜利，-1表示左边的队伍失败，0表示平均，胜利的队得3分，失败的队不得分，平均各得一分。

思路

一共n(n-1)/2场比赛，最多就n(n-1)/2个场次有队伍可以胜利，平摊给每一个队就是(n-1)/2,但是可能存在(n-1)%2==1，那么就必须每个队伍需要平局，否则，每个队胜利(n-1)/2即可，其余n-1场失败就可以了。对于平均可以1~2、3~4、5~6…平局。假设平均胜利场次pwin=(n-1)/2,平均失败场次plose=n-1-pwin-pp(pp表示是否存平局，0表示不存在，1表示存在)，对于每次枚举的主场队伍i，如果它还需要胜利，并且对手j还需要失败，就让i胜利，否则就让它失败。

代码

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn=1e5+50;
int T,n,win[maxn],lose[maxn];
int main(){
    cin>>T;
    while(T--){
        cin>>n;
        memset(win,0,sizeof(int)*(n+10));
        memset(lose,0,sizeof(int)*(n+10));
        int tot=(n-1)*n/2;
        int pp=(tot%n)*2/n;
        int pwin=tot/n;
        int plose=n-1-pwin-pp;
        int sign=0;
        if(pp)sign=1;
        for(int i=1;i<=n;i++){
            for(int j=i+1;j<=n;j++){
                if(sign&&i%2){
                    if(j==i+1){
                        cout<<0<<' ';
                    }
                    else{
                        if(win[i]<pwin&&lose[j]<plose){
                            cout<<1<<' ';
                            win[i]++;lose[j]++;
                        }
                        else cout<<-1<<' ';
                    }
                }
                else{
                    if(win[i]<pwin&&lose[j]<plose){
                            cout<<1<<' ';
                            win[i]++;lose[j]++;
                        }
                        else cout<<-1<<' ';
                }
            }
        }

        cout<<endl;
    }
    return 0;
}

D. Pythagorean Triples

题目大意

问1<=a<=b<=c<=n中，有多少个(a,b,c)满足

思路

将(1)完全平方得：

和(2) 联立得：

所以：

而

注意条件1<=a<=b<=c<=n,并且a a -1和a a+1要是偶数。直接枚举a（1<=a<=sqrt(2*n+1)）

代码

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll T,n;
int main(){
    cin>>T;
    while(T--){
        cin>>n;
        ll ans=0;
        ll up=sqrt(2*n+1);
        for(ll a=1;a<=up;a++){
            if((a*a-1)%2)continue;
            ll b=(a*a-1)/2;
            if(b<a||b>n)continue;
            if((a*a+1)%2)continue;
            ll c=(a*a+1)/2;
            if(c<b||c>n)continue;
            ans++;
        }
        cout<<ans<<endl;
    }    
    return 0;
}

E. Cheap Dinner

题目大意

有n1个first courses,n2个second courses，n3个 drinks，n4个 desserts，a[],b[],c[],d[]分别对应价格，m1组xi，yi表示第xi种first courses和第y种second courses不可搭配，m2、m3组xi，yi同理。问搭配出合理套餐的最低价，如果不能搭配就输出-1.

思路

对于n2个second courses，它最多就只有m1对first courses不能搭配，如果我们找到每一个second courses搭配出first courses的最低价由此更新second courses，即bi=bi+min{ai}，先将a[]从小到大排好，然后用vector储存second courses不能搭配的first courses，然后对vectors[i]排序，找到第i个second courses可以搭配的最便宜的first courses。同理对于drinks、desserts亦可以用上述的办法，ci=ci+min{bi},di=di+min{ci};显然min{di}就是答案

代码

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn=200000+50;
ll T,n1,n2,n3,n4,m1,m2,m3,id[maxn];vector<int>s[maxn];
struct Node{
    ll w,id;
}a[maxn],b[maxn],c[maxn],d[maxn];
bool cmp(Node x,Node y){
    if(x.w==y.w)return x.id<y.id;
    return x.w<y.w;
}
int main(){
    cin>>n1>>n2>>n3>>n4;
    for(int i=1;i<=n1;i++)cin>>a[i].w,a[i].id=i;
    for(int i=1;i<=n2;i++)cin>>b[i].w,b[i].id=i;
    for(int i=1;i<=n3;i++)cin>>c[i].w,c[i].id=i;
    for(int i=1;i<=n4;i++)cin>>d[i].w,d[i].id=i;
    sort(a+1,a+1+n1,cmp);
    for(int i=1;i<=n1;i++)id[a[i].id]=i;
    cin>>m1;
    for(int i=1;i<=m1;i++){
        int x,y;cin>>x>>y;
        s[y].push_back(id[x]);
    }
    for(int i=1;i<=n2;i++){
        int opt=1;
        sort(s[i].begin(),s[i].end());
        for(int j=0;j<s[i].size();j++){
            if(s[i][j]==opt){
                opt++;
            }
            else{
                break;
            }
        }
        if(opt==n1+1)b[i].w=1e9;
        else b[i].w+=a[opt].w;
    }
    sort(b+1,b+1+n2,cmp);
    for(int i=1;i<=n2;i++)id[b[i].id]=i;
    cin>>m2;
    for(int i=1;i<=150000;i++)s[i].clear();
    for(int i=1;i<=m2;i++){
        int x,y;cin>>x>>y;
        s[y].push_back(id[x]);
    }
    for(int i=1;i<=n3;i++){
        int opt=1;
        sort(s[i].begin(),s[i].end());
        for(int j=0;j<s[i].size();j++){
            if(s[i][j]==opt){
                opt++;
            }
            else{
                break;
            }
        }
        if(opt==n2+1)c[i].w=1e9;
        else c[i].w+=b[opt].w;
        if(c[i].w>=1e9)c[i].w=1e9;
    }
    sort(c+1,c+1+n3,cmp);
    for(int i=1;i<=n3;i++)id[c[i].id]=i;
    cin>>m3;
    for(int i=1;i<=150000;i++)s[i].clear();
    for(int i=1;i<=m3;i++){
        int x,y;cin>>x>>y;
        s[y].push_back(id[x]);
    }
    for(int i=1;i<=n4;i++){
        int opt=1;
        sort(s[i].begin(),s[i].end());
        for(int j=0;j<s[i].size();j++){
            if(s[i][j]==opt){
                opt++;
            }
            else{
                break;
            }
        }
        if(opt==n3+1)d[i].w=1e9;
        else d[i].w+=c[opt].w;
        if(d[i].w>=1e9)d[i].w=1e9;    
    }
    ll ans=1e9;
    for(int i=1;i<=n4;i++){
        ans=min(ans,d[i].w);
    }
    if(ans==1e9)cout<<-1<<endl;
    else cout<<ans<<endl;
    return 0;
}
